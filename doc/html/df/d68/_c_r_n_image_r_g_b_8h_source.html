82  template<typename T> Image<T> YChannel(const Image<pixel::RGB<T>> &img)
83  {
84  auto out = Image<T>{img.GetWidth(), img.GetHeight()};
85  for (auto tmp : Range(img))
86  out.At(tmp) = T(DecimalType<T>(img.At(tmp).r) * 0.299 +
87  DecimalType<T>(img.At(tmp).g) * 0.587 +
88  DecimalType<T>(img.At(tmp).b) * 0.114);
89  return out;
90  }

92  template<typename T> Image<T> LChannel(const Image<pixel::RGB<T>> &img)
93  {
94  auto out = Image<T>{img.GetWidth(), img.GetHeight()};
95  for (auto tmp : Range(img))
96  out.At(tmp) = T((Max(img.At(tmp).r, img.At(tmp).g, img.At(tmp).b) +
97  Min(img.At(tmp).r, img.At(tmp).g, img.At(tmp).b)) / 2);
98  return out;
99  }
