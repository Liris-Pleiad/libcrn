66  _("Negative standard deviation"));
67  if (sigma <= 0.001)

89 MatrixDouble MatrixDouble::NewGaussianLineDerivative(double sigma)
90 {
91  if (sigma < 0)
92  throw ExceptionDomain(StringUTF8("MatrixDouble* MatrixDouble::NewGaussianLineDerivative(double sigma): ") +
93  _("Negative standard deviation"));
94  if (sigma == 0)
95  {
96  MatrixDouble mat(1, 3, 0);
97  mat[0][0] = 1;
98  mat[0][2] = -1;
99  return mat;
100  }
101 
102  int hwin = (int)ceil(Thrice(sigma));
103  int size = hwin + hwin + 1;
104  MatrixDouble mat(1, size, 0);
105 
106  for (int tmp = 0; tmp < size; ++tmp)
107  {
108  double v = (double)hwin - (double)tmp; // -x
109  mat.At(0, tmp) = (v * Gauss(v, sigma) +
110  (v - 0.5) * Gauss(v - 0.5, sigma) +
111  (v + 0.5) * Gauss(v + 0.5, sigma)) / 3.0;
112  }
113 
114  return mat;
115 }

126 MatrixDouble MatrixDouble::NewGaussianLineSecondDerivative(double sigma)
127 {
128  if (sigma < 0)
129  throw ExceptionDomain(StringUTF8("MatrixDouble* MatrixDouble::NewGaussianLineSecondDerivative(double sigma): ") +
130  _("Negative standard deviation"));
131  if (sigma == 0)
132  {
133  MatrixDouble mat(1, 3, -1);
134  mat[0][1] = 2;
135  return mat;
136  }
137  int hwin = (int)ceil(6 * sigma);
138  int size = hwin + hwin + 1;
139  MatrixDouble mat(1, size, 0);
140  for (int tmp = 0; tmp < size; ++tmp)
141  {
142  double v = (double)tmp - (double)hwin; // x
143  mat.At(0, tmp) = ((Sqr(v) - Sqr(sigma)) * Gauss(v, sigma) +
144  (Sqr(v - 0.5) - Sqr(sigma)) * Gauss(v - 0.5, sigma) +
145  (Sqr(v + 0.5) - Sqr(sigma)) * Gauss(v + 0.5, sigma)) / 3.0;
146  }
147  return mat;
148 }
