564 {

715 int String::EditDistance(const String &s) const
716 {
717  unsigned int s1 = (unsigned int)Size();
718  unsigned int s2 = (unsigned int)s.Size();
719  std::vector<std::vector<unsigned int> > d(s1 + 1, std::vector<unsigned int>(s2 + 1));
720 
721  for (unsigned int i = 1; i <= s1; ++i) d[i][0] = i;
722  for (unsigned int i = 1; i <= s2; ++i) d[0][i] = i;
723 
724  for (unsigned int i = 1; i <= s1; ++i)
725  for (unsigned int j = 1; j <= s2; ++j)
726  d[i][j] = std::min(std::min(d[i - 1][j] + 1, d[i][j - 1] + 1),
727  d[i - 1][j - 1] + (data[i - 1] == s[j - 1] ? 0 : 1));
728  return d[s1][s2];
729 }

774 {
775  size_t n = data.length();
