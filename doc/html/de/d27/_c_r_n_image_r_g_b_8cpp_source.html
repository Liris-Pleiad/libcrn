64 static std::pair<bool, String> save_png_libpng(const Path &filename, const ImageRGB &img)
65 {
66  // libpng does not support URIs
67  Path fname(filename);
68  fname.ToLocal();
69 
70  std::unique_ptr<FILE, decltype(fclose_if_not_null)*> fp(fopen(fname.CStr(), "wb"), fclose_if_not_null);
71  if (!fp)
72  {
73  return std::make_pair(false, String(_("Cannot create file ")) + U"<" + fname + U">");
74  }
75 
76  /* Create png struct & info */
77  png_structp png_ptr = png_create_write_struct (PNG_LIBPNG_VER_STRING, NULL, NULL, NULL);
78  if (!png_ptr)
79  {
80  return std::make_pair(false, String(_("Cannot create the PNG structure.")));
81  }
82  png_infop info_ptr = png_create_info_struct(png_ptr);
83  if (!info_ptr)
84  {
85  png_destroy_write_struct(&png_ptr, (png_infopp)NULL);
86  return std::make_pair(false, String(_("Cannot create the PNG info.")));
87  }
88 
89  /* Set up the error handling */
90  if (setjmp(png_jmpbuf(png_ptr)))
91  {
92  png_destroy_write_struct(&png_ptr, &info_ptr);
93  return std::make_pair(false, String(_("Error while generating the PNG image.")));
94  }
95 
96  /* setup libpng for using standard C fwrite() function with our FILE pointer */
97  png_init_io(png_ptr, fp.get());
98 
99  /* Fill in the png_info structure */
100  int width = int(img.GetWidth());
101  int height = int(img.GetHeight());
102  int bit_depth = 8;
103  int color_type = PNG_COLOR_TYPE_RGB;
104  int interlace_type = PNG_INTERLACE_NONE;
105  int compression_type = PNG_COMPRESSION_TYPE_DEFAULT;
106  int filter_method = PNG_FILTER_TYPE_DEFAULT;
107  png_set_IHDR(png_ptr, info_ptr, width, height, bit_depth, color_type, interlace_type, compression_type, filter_method);
108 
109  /* Using the low-level write interface */
110  png_write_info(png_ptr, info_ptr);
111  png_bytepp row_pointers;
112  row_pointers = (png_bytep*)malloc(sizeof(png_byte*) * height);
113  for (int i = 0; i < height ; ++i)
114  {
115 #if (PNG_LIBPNG_VER > 10300)
116  row_pointers[i] = (png_bytep)calloc(1, png_get_rowbytes(png_ptr, info_ptr));
117 #else
118  row_pointers[i] = (png_bytep)calloc(1, info_ptr->rowbytes);
119 #endif
120  }
121 
122  /* Conversion */
123 #if (PNG_LIBPNG_VER > 10300)
124  int channels = png_get_channels(png_ptr, info_ptr);
125 #else
126  int channels = info_ptr->channels;
127 #endif
128  for (int y = 0; y < height; y++)
129  for (int x = 0; x < width; x++)
130  {
131  const auto px = img.At(x, y);
132  row_pointers[y][x * channels] = px.r;
133  row_pointers[y][x * channels + 1] = px.g;
134  row_pointers[y][x * channels + 2] = px.b;
135  }
136 
137  png_write_image(png_ptr, row_pointers);
138 
139  // TODO more options
140  png_text txt[1];
141  txt[0].compression = PNG_TEXT_COMPRESSION_NONE;
142  txt[0].key = (char*)"creator";
143  txt[0].text = (char*)"libcrn";
144  txt[0].text_length = strlen(txt[0].text);
145  png_set_text(png_ptr, info_ptr, txt, 1);
146 
147  png_write_end(png_ptr, info_ptr);
148  png_destroy_write_struct(&png_ptr, &info_ptr);
149 
150  // free
151  for (int i = 0; i < height ; ++i)
152  {
153  free(row_pointers[i]);
154  }
155  free(row_pointers);
156  return std::make_pair(true, String(""));
157

323 {
324  // libjpeg does not support URIs
325  Path fname(filename);
326  fname.ToLocal();
327 
328  std::unique_ptr<FILE, decltype(fclose_if_not_null)*> fp(fopen(fname.CStr(), "wb"), fclose_if_not_null);
329  if (!fp)
330  {
331  return std::make_pair(false, String(_("Cannot create file ")) + U"<" + fname + U">");
332  }
333 
334  crn_jpeg_error_mgr jerr;
335  struct jpeg_compress_struct cinfo;
336 
337  cinfo.err = jpeg_std_error(&jerr.pub);
338  jerr.pub.error_exit = crn_jpeg_error_exit;
339  if (setjmp(jerr.setjmp_buffer))
340  {
341  jpeg_destroy_compress(&cinfo);
342  return std::make_pair(false, String(_("Cannot create jpeg file structure.")));

344 
345  jpeg_create_compress(&cinfo);
346  jpeg_stdio_dest(&cinfo, fp.get());
347  cinfo.image_width = int(img.GetWidth());
348  cinfo.image_height = int(img.GetHeight());
349  cinfo.input_components = 3;
350  cinfo.in_color_space = JCS_RGB;
351  jpeg_set_defaults(&cinfo);
352  jpeg_set_quality(&cinfo, qual, TRUE);
353  jpeg_start_compress(&cinfo, TRUE);
354  JSAMPROW scanline;
355  scanline = new JSAMPLE[3 * img.GetWidth()];
356  for (size_t y = 0; y < img.GetHeight(); ++y)
357  {
358  for (size_t x = 0; x < img.GetWidth(); ++x)
359  {
360  const auto &p = img.At(x, y);
361  scanline[3 * x] = p.r;
362  scanline[3 * x + 1] = p.g;
363  scanline[3 * x + 2] = p.b;
364  }
365  jpeg_write_scanlines(&cinfo, &scanline, 1);
366  }
367  delete scanline;
368  // write comment
369  jpeg_write_marker(&cinfo, JPEG_COM, (JOCTET*)"Saved by libcrn.", (unsigned int)strlen("Saved by libcrn."));
370 
371  jpeg_finish_compress(&cinfo);
372  jpeg_destroy_compress(&cinfo);
373 
374  return std::make_pair(true, String(U""));
375 }

526 void crn::EdgePreservingFilter(ImageRGB &img, size_t iter, uint8_t maxdist)
527 {
528  if (iter == 0)
529  throw ExceptionInvalidArgument(StringUTF8("EdgePreservingFilter(ImageRGB &img, size_t iter, uint8_t maxdist): ") + _("The number of iterations is null."));
530  // buffer
531  auto newpix = ImageRGB(img.GetWidth(), img.GetHeight());
532  // reference buffer
533  auto refpix = img;
534 
535  const int lin[8] = {-1,-1,-1, 0, 1, 1, 1, 0};
536  const int col[8] = {-1, 0, 1, 1, 1, 0,-1,-1};
537  for (size_t tmp = 0; tmp < iter; ++tmp)
538  { // for each iter
539  for (int y = 0; y < int(img.GetHeight()); ++y)
540  for (int x = 0; x < int(img.GetWidth()); ++x)
541  {
542  int tr = 0, tg = 0, tb = 0, mr = 0, mg = 0, mb = 0;
543  for (size_t d = 0; d < 8; ++d)
544  { // for each neighbour
545  int xx = x + col[d];
546  int yy = y + lin[d];
547  if ((xx >= 0) && (xx < int(img.GetWidth())) && (yy >= 0) && (yy < int(img.GetHeight())))
548  {
549  int dr = Max(0, 1 - Abs(refpix.At(x, y).r - img.At(xx, yy).r) / maxdist);
550  tr += dr;
551  mr += dr * img.At(xx, yy).r;
552  int dg = Max(0, 1 - Abs(refpix.At(x, y).g - img.At(xx, yy).g) / maxdist);
553  tg += dg;
554  mg += dg * img.At(xx, yy).g;
555  int db = Max(0, 1 - Abs(refpix.At(x, y).b - img.At(xx, yy).b) / maxdist);
556  tb += db;
557  mb += db * img.At(xx, yy).b;
558  }
559  }
560  // update buffer
561  if (tr) mr /= tr;
562  newpix.At(x, y).r = uint8_t(mr);
563  if (tg) mg /= tg;
564  newpix.At(x, y).g = uint8_t(mg);
565  if (tb) mb /= tb;
566  newpix.At(x, y).b = uint8_t(mb);
567  }
568  // swap buffer and reference buffer
569  std::swap(newpix, refpix);
570  }
571  img = std::move(refpix);
572 }

585 ImageGray crn::SChannel(const ImageRGB &img)
586 {

611  for (size_t tmp = 0; tmp < img.Size(); tmp++)
612  {
613  const auto px = img.At(tmp);
614  int max = Max(px.r, px.g, px.b);
615  int min = Min(px.r, px.g, px.b);
616  auto sat = 0.0;
617  if (max > 0)
618  sat = 1.0 - ((double)min / max);
619  auto s = (int)(sat * 255);
620  auto v = int(g.At(tmp));
621  v *= 255 - s;
622  v /= 255;
623  g.At(tmp) = (uint8_t)Cap(v, 0, 255);
624  }
625  return g;
626 }
