181  template<typename ITER> unsigned int UnivariateGaussianMixture::EM(ITER it_begin, ITER it_end, size_t nbSeeds, double epsilon, size_t maximalIterations)
182  {
183  unsigned int nbIterations = 0;
184  size_t nbMembers = size_t(nbSeeds);
185  size_t nbPatterns = 0;
186  size_t nbKeys = std::distance(it_begin, it_end);
187 
189  // Setup //
191 
192  auto min_value = std::numeric_limits<double>::infinity();
193  auto max_value = -std::numeric_limits<double>::infinity();
194 
195  for (auto it = it_begin; it != it_end; ++it)
196  {
197  auto v = it->first;
198 
199  min_value = Min(min_value, v);
200  max_value = Max(max_value, v);
201  nbPatterns += it->second;
202  }
203 
204  members.clear();
205 
206 
207  double delta = (max_value - min_value) / double(nbSeeds);
208  double Seed = min_value + delta / 2.0;
209 
210  // Heuristic setup : gaussians are placed equidistant from Min to Max value
211  // variance set to catch at least one data value
212 
213  for (size_t k = 0; k < nbMembers; ++k)
214  {
215  // Warning : if data distribution is too sparse, variance should be adapted
216 
217  bool redo = true;
218  double var = delta;
219  double sigma = sqrt(var);
220 
221  auto it = it_begin;
222 
223  while (redo)
224  {
225  redo = (Abs(Seed - it->first) > sigma);
226  ++it;
227 
228  if ((it == it_end) && redo)
229  {
230  var += delta;
231  sigma = sqrt(var);
232  it = it_begin;
233  }
234  }
235 
236  AddMember(UnivariateGaussianPDF(Seed, var), 1.0);
237  Seed += delta;
238  }
239 
240  // Heuristic setup : mobile centroids
241 
242 
243 
244  MatrixDouble Proba(nbKeys, nbMembers, 0.0);
245 
246  bool Continue = true;
247  double likelihood = 0.0;
248 
250  // Iterations //
252 
253  while(Continue)
254  {
255  // Expectation ("E" step)
256 
257  auto it = it_begin;
258 
259  for (size_t i = 0; i < nbKeys; ++i)
260  {
261  double xi = it->first;
262  double Pi = 0.0;
263 
264  for (size_t k = 0; k < nbMembers; ++k)
265  {
266  double pik = this->ValueAt(xi, k);
267 
268  Proba[i][k] = pik;
269  Pi += pik;
270  }
271 
272  Proba.MultRow(i, 1.0 / Pi);
273  ++it;
274  }
275 
276  // Maximization ("M" step)
277 
278  for (size_t k = 0; k < nbMembers; ++k)
279  {
280  const auto& Gk = members[k].first;
281  double CumulPk = 0.0;
282  double muk = Gk.GetMean();
283 
284  double mu = 0.0;
285  double v = 0.0;
286 
287  it = it_begin;
288 
289  for (size_t i = 0; i < nbKeys; ++i)
290  {
291  double pik = Proba[i][k];
292  double xi = it->first;
293  double ni = double(it->second);
294 
295  CumulPk += ni * pik;
296  mu += ni * pik * xi;
297  v += ni * pik * Sqr(xi - muk);
298  ++it;
299  }
300 
301  if (std::isfinite(mu) && std::isfinite(v))
302  {
303  mu /= CumulPk;
304  v /= CumulPk;
305  }
306  else // Numeric limit reached. Recomputation needed
307  {
308  it = it_begin;
309  mu = 0.0;
310  v = 0.0;
311 
312  for (size_t i = 0; i < nbKeys; ++i)
313  {
314  double pik = Proba[i][k];
315  double xi = it->first;
316  double ni = double(it->second);
317 
318  mu += ni * pik * xi / CumulPk;
319  v += ni * pik * Sqr(xi - muk) / CumulPk;
320  ++it;
321  }
322  }
323 
324  members[k].second = CumulPk / double(nbPatterns);
325  members[k].first = UnivariateGaussianPDF(mu, v);
326  }
327 
328  double NewLikelihood = MLLE(it_begin, it_end);
329  double LikelihoodDiff = fabs(NewLikelihood - likelihood);
330 
331  likelihood = NewLikelihood;
332  ++nbIterations;
333 
334  Continue = ((nbIterations < maximalIterations) && (LikelihoodDiff > epsilon));
335  }
336 
337  return nbIterations;
338  }
