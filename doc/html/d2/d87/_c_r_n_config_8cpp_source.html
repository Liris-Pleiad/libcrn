36 Config::Config():
37  conf("crn")
38 {
39  if (!conf.Load())
40  {
41  // default init
42  conf.SetData(topDirKey(), Path(CRN_PROJECT_FULL_PATH));
43  conf.SetData(localeDirKey(), Path(CRN_LOCALE_FULL_PATH));
44  conf.SetData(staticDataDirKey(), Path(CRN_DATA_FULL_PATH));
45  conf.SetData(verboseKey(), Prop3::True);
46  conf.Save();
47  }
48  IO::IsVerbose() = conf.GetProp3(verboseKey()).IsTrue();
49  // misc init
50 #ifdef CRN_USING_GDKPB
51 # if !GLIB_VERSION_2_36
52  g_type_init();
53 # endif
54 #endif
55 
56  // i18n
57  char *loc = getenv("LANG");
58  CRNdout << "libcrn LANG env: " << (loc == nullptr ? "none" : loc) << std::endl;
59 
60  loc = setlocale(LC_ALL, "");
61 
62  if (!loc)
63  CRNdout << "libcrn setlocale failed" << std::endl;
64  else
65  CRNdout << "libcrn locale = " << loc << std::endl;
66 
67  loc = CRNbindtextdomain(GETTEXT_PACKAGE, conf.GetPath(localeDirKey()).CStr());
68  if (!loc)
69  CRNdout << "libcrn: no bound path. should be " << conf.GetPath(localeDirKey()).CStr() << std::endl;
70  else
71  CRNdout << "libcrn: path set to " << loc << std::endl;
72  loc = CRNbind_textdomain_codeset(GETTEXT_PACKAGE, "UTF-8");
73  if (!loc)
74  CRNdout << "libcrn: no bound codeset. should be UTF-8" << std::endl;
75  else
76  CRNdout << "libcrn: Codeset = " << loc << std::endl;
77  loc = CRNtextdomain(GETTEXT_PACKAGE);
78  if (!loc)
79  CRNdout << "libcrn textdomain failed" << std::endl;
80  else
81  CRNdout << "libcrn text domain = " << loc << std::endl;
82  CRNdout << _("Using default language.") << std::endl;
83 }
