609 Rect::spiral_iterator::spiral_iterator(const Rect &r):
610  rectl(r.GetLeft()),
611  rectt(r.GetTop()),
612  rectr(r.GetRight()),
613  rectb(r.GetBottom()),
614  ref(r.GetCenterX(), r.GetCenterY()),
615  dir(((r.GetWidth() >= r.GetHeight()) ? Direction::RIGHT : Direction::TOP)),
616  pass(0),
617  preproc(false),
618  valid(true)
619 {
620  pos = ref;
621  border = Min(ref.X - r.GetLeft(), ref.Y - r.GetTop());
622  framel = rectl + border;
623  framet = rectl + border;
624  framer = rectr - border;
625  frameb = rectb - border;
626  if (((framel == framer) || (framet == frameb)) && (r.GetWidth() != r.GetHeight()))
627  {
628  preproc = true;
629  if (framel == framer)
630  {
631  pdir = Point2DInt(0, 1);
632  offset = pos.Y - r.GetTop();

635  pass = 1;

640  offset = pos.X - r.GetLeft();
641  limit = pos.Y - r.GetTop() - 1;

643  pass = 1;
644  }
645  }

741 void Rect::spiral_iterator::update()
742 {
743  if (pos == ref)
744  {
745  border -= 1;
746  if (border < 0)
747  {
748  valid = false;
749  return;
750  }
751 
752  framel = rectl + border;
753  framet = rectt + border;
754  framer = rectr - border;
755  frameb = rectb - border;
756  if (dir == Direction::RIGHT)
757  pos.Y -= 1;
758  else if (dir == Direction::BOTTOM)
759  pos.X += 1;

761  pos.Y += 1;
762  else if (dir == Direction::TOP)
763  pos.X -= 1;
764  ref = pos;
