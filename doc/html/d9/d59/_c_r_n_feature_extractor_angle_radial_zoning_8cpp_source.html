34 static inline size_t chooseangle(const crn::Angle<ByteAngle> &a, const std::vector<crn::Angle<ByteAngle> > &angles)
35 {
36  std::vector<uint8_t> distances(angles.size());
37  for (size_t tmp = 0; tmp < angles.size(); ++tmp)
38  distances[tmp] = crn::AngularDistance(a, angles[tmp]);
39  return std::min_element(distances.begin(), distances.end()) - distances.begin();
40 }

49 SObject FeatureExtractorAngleRadialZoning::Extract(Block &b)
50 {
51  SImageGray ig(b.GetGray());
52  size_t xbar, ybar;
53  if (autocenter)
54  {
55  size_t m00 = 0;
56  xbar = 0;
57  ybar = 0;
58  FOREACHPIXEL(x, y, *ig)
59  {
60  int val = 255 - ig->At(x, y);
61  m00 += val;
62  xbar += x * val;
63  ybar += y * val;
64  }
65  if (m00)
66  {
67  xbar /= m00;
68  ybar /= m00;
69  }
70  else
71  {
72  xbar = ig->GetWidth() / 2;
73  ybar = ig->GetHeight() / 2;
74  }
75  }
76  else
77  {
78  xbar = ig->GetWidth() / 2;
79  ybar = ig->GetHeight() / 2;
80  }
81 
82  // create feature vector
83  auto feat = std::make_shared<Vector>();
84  for (size_t tmp = 0; tmp < nbzones; ++tmp)
85  feat->PushBack(std::make_shared<Histogram>(zonesize));
86  // extract angles
87  SImageGradient igr = b.GetGradient();
88  unsigned int cnt = 0;
89  FOREACHPIXEL(x, y, *ig)
90  {
91  if (igr->IsSignificant(x, y))
92  {
93  size_t octant = chooseangle(crn::Angle<ByteAngle>::Atan(double(y) - double(ybar), double(x) - double(xbar)), zonetab);
94  SHistogram h(std::static_pointer_cast<Histogram>((*feat)[octant]));
95  h->IncBin(chooseangle(igr->At(x, y).theta, feattab));
96  cnt += 1;
97  }
98  }
99  // normalize
100  if (cnt)
101  {
102  for (SObject oh : feat)
103  {
104  SHistogram h(std::static_pointer_cast<Histogram>(oh));
105  for (size_t tmp = 0; tmp < zonesize; ++tmp)
106  h->SetBin(tmp, (h->GetBin(tmp) * maxval) / cnt);
107  }
108  }
109  return feat;
110 }
