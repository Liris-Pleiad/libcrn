138 MatrixDouble GaussianSCHMM::alpha(const MatrixDouble& Observed) const
139 {
140  size_t i, j, t;
141  size_t N = nbStates;
142  size_t T = Observed.GetRows();
143 
144  MatrixDouble FWD(T, N, 0.0);
145  MatrixDouble A = stateTransitionProbability;
146  const std::vector<MultivariateGaussianMixture> &B = stateGivenSymbolProbability;
147  MatrixDouble P = firstStateProbability;
148 
149  // First alpha for each state.
150 
151  for (i = 0; i < N; i++)
152  {
153  FWD[0][i] = P.At(i, 0) * B[i].ValueAt(Observed.MakeRowAsColumn(0));
154  }
155 
156  // Iteration.
157 
158  for (t = 0; t < T - 1; t++)
159  {
160  for (j = 0; j < N; j++)
161  {
162  double Sum = 0.0;
163 
164  for (i = 0; i < N; i++)
165  Sum += FWD[t][i] * A[i][j];
166 
167  FWD[t + 1][j] = Sum * B[j].ValueAt(Observed.MakeRowAsColumn(t + 1));
168  }
169  }
170 
171  return std::forward<MatrixDouble>(FWD);
172 }

181 MatrixDouble GaussianSCHMM::beta(const MatrixDouble& Observed) const
182 {
183  size_t i,j;
184  size_t N = nbStates;
185  size_t T = Observed.GetRows();
186 
187  MatrixDouble BKD(T, N, 0.0);
188  MatrixDouble A = stateTransitionProbability;
189  const std::vector<MultivariateGaussianMixture> &B = stateGivenSymbolProbability;
190 
191  // Last beta for each state.
192 
193  for (i = 0; i < N; i++)
194  BKD[T - 1][i] = 1.0;
195 
196  // Iteration.
197 
198  for (int t = int(T) - 2; t >= 0; t--)
199  {
200  for (i = 0; i < N; i++)
201  {
202  double Sum = 0.0;
203 
204  for (j = 0; j < N; j++)
205  Sum += A[i][j] * B[j].ValueAt(Observed.MakeRowAsColumn(t + 1)) * BKD[t + 1][j];
206 
207  BKD[t][i] = Sum;
208  }
209  }
210 
211  return std::forward<MatrixDouble>(BKD);
212 }

220 void GaussianSCHMM::BaumWelchSingle(const MatrixDouble& Observed, unsigned int MaxIter)
221 {
222  unsigned int Count = 1;
223  bool Continue = true;
224 
225  while (Continue)
226  {
227  // One step of Baum-Welch algorithm
228 
229  size_t i, j, k, t;
230  size_t N = nbStates;
231  size_t T = Observed.GetRows();
232 
233  // Some variables to make code clearer.
234 
235  SquareMatrixDouble A = stateTransitionProbability;
236  std::vector<MultivariateGaussianMixture> &B = stateGivenSymbolProbability;
237  MatrixDouble P = firstStateProbability;
238 
239  // Useful numerical table
240  // - forward function
241  // - backward function
242  // - Xi (needed for transition reestimation)
243  // - Gamma (needed for transition reestimation)
244 
245  MatrixDouble Alpha(alpha(Observed));
246  MatrixDouble Beta(beta(Observed));
247  std::vector<SquareMatrixDouble> Xi;
248  MatrixDouble Gamma(T, N, 0.0);
249 
250  for (k = 0; k < T; k++)
251  Xi.push_back(SquareMatrixDouble(N));
252 
253  // Compute sequence probability via Alpha
254  // (needed for transition reestimation)
255 
256  double Prob = 0.0;
257 
258  for (k = 0; k < N; k++)
259  Prob += Alpha[T - 1][k];
260 
261  // Computation for Xi and Gamma
262 
263  for (t = 0; t < T - 1; t++)
264  {
265  for (i = 0; i < N; i++)
266  {
267  for (j = 0; j < N; j++)
268  {
269  double z = (Alpha[t][i] * A[i][j] * B[j].ValueAt(Observed.MakeRowAsColumn(t + 1)) * Beta[t + 1][j]) / Prob;
270  Xi[t][i][j] = z;
271  Gamma[t][i] += z;
272  }
273  }
274  }
275 
276  // Last gamma vector (cf. Rabiner & Juang).
277 
278  for (i = 0; i < N; i++)
279  {
280  double alpha = Alpha[T - 1][i];
281  double beta = Beta[T - 1][i];
282  double g = alpha * beta / Prob;
283 
284  Gamma[T - 1][i] += g;
285  }
286 
287  // Re-estimation
288 
289  // Re-estimation for P.
290 
291  P = MatrixDouble(Gamma.MakeRowAsColumn(0));
292 
293  firstStateProbability = P;
294 
295  // Re-estimation for A.
296 
297  A.SetAll(0.0);
298 
299  MatrixDouble InvScales(N, 1, 0.0);
300 
301  for (t = 0; t < T - 1; t++)
302  {
303  for (i = 0; i < N; i++)
304  {
305  InvScales[i][0] += Gamma[t][i];
306 
307  for (j = 0; j < N; j++)
308  A[i][j] += Xi[t].At(i, j);
309  }
310  }
311 
312  for (i = 0; i < N; i++)
313  {
314  for (j = 0; j < N; j++)
315  {
316  double v = A.At(i, j);
317 
318  A[i][j] = v / InvScales[i][0];
319  }
320  }
321 
322  stateTransitionProbability = A;
323 
324  // Re-estimation for B :
325  //
326  // - reestimation for mixture weights
327  // - reestimation for means
328  // - reestimation for variances
329 
330  // The "new" gamma in Rabiner's
331 
332  //std::vector<std::vector<MatrixDouble>> ExtendedGamma(T, std::vector<MatrixDouble>(N));
333  std::vector< std::vector< std::vector<double> > > ExtendedGamma(T, std::vector< std::vector<double> >(N));
334 
335  for (t = 0; t < T; t++)
336  {
337  MatrixDouble Ot = Observed.MakeRowAsColumn(t);
338 
339  for (j = 0; j < N; j++)
340  {
341  size_t Mj = B[j].GetNbMembers(); // Number of members in j-th mixture
342 
343  //MatrixDouble V(Mj, 1, 0.0);
344  std::vector<double> V(Mj);
345 
346  double Denominator = B[j].ValueAt(Ot);
347 
348  for (k = 0; k < Mj; k++)
349  {
350  double Numerator = B[j].ValueAt(Ot, k);
351 
352  V[k] = Gamma[t][j] * Numerator / Denominator;
353  }
354 
355  ExtendedGamma[t][j] = V;
356  }
357  }
358 
359  for (j = 0; j < N; j++)
360  {
361  size_t Mj = B[j].GetNbMembers();
362 
363  for (k = 0; k < Mj; k++)
364  {
365  double c_jk = 0.0;
366  MatrixDouble mu_jk(symbolDimension, 1, 0.0);
367  SquareMatrixDouble U_jk(symbolDimension, 0.0);
368 
369  // k-th distribution in j-th mixture
370  MultivariateGaussianPDF Gjk = B[j].GetMember(k);
371  MatrixDouble mu_jk_old = Gjk.GetMean();
372 
373  // Scaling factors
374  double c_jk_scale = 0.0;
375  double mu_jk_scale = 0.0;
376  double U_jk_scale = 0.0;
377 
378  for (t = 0; t < T; t++)
379  {
380  MatrixDouble Ot = Observed.MakeRowAsColumn(t);
381  //double gamma_tjk = ExtendedGamma[t][j].At(k, 0);
382  double gamma_tjk = ExtendedGamma[t][j][k];
383 
384  // increase c_jk
385  c_jk += gamma_tjk;
386 
387  // increase mu_jk
388  //mu_jk.add(Ot.getScaled(gamma_tjk));
389  MatrixDouble tm = Ot;
390  tm *= gamma_tjk;
391  mu_jk += tm;
392 
393  // increase U_jk
394  //SMatrixDouble* TranslatedPattern = Ot.getSubstracted(mu_jk_old);
395  MatrixDouble TranslatedPattern = Ot;
396  TranslatedPattern -= mu_jk_old;
397  //U_jk.add(TranslatedPattern.getRightAutoProduct().getScaled(gamma_tjk));
398  SquareMatrixDouble tsm = TranslatedPattern.MakeVectorRightAutoProduct();
399  tsm *= gamma_tjk;
400  U_jk += tsm;
401 
402  // Increase scale factors
403 
404  for (size_t m = 0; m < Mj; m++)
405  //c_jk_scale += ExtendedGamma[t][j].At(m, 0);
406  c_jk_scale += ExtendedGamma[t][j][m];
407 
408  mu_jk_scale += gamma_tjk;
409  U_jk_scale += gamma_tjk;
410  }
411 
412  c_jk /= c_jk_scale;
413  mu_jk *= 1.0 / mu_jk_scale;
414  U_jk *= 1.0 / U_jk_scale;
415 
416  stateGivenSymbolProbability[j].SetMember(MultivariateGaussianPDF(mu_jk, U_jk), c_jk, k);
417  }
418  }
419 
420  ExtendedGamma.clear();
421 
422  Count++;
423  //Continue = !((this.equivalent(H)) || (Count > MaxIter));
424  Continue = !(Count > MaxIter);
425  }
426 }
