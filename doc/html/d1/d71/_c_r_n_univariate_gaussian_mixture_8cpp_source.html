313 unsigned int UnivariateGaussianMixture::EM(const MatrixDouble& patterns, size_t nbSeeds, double epsilon, size_t maximalIterations)
314 {
315  unsigned int nbIterations = 0;
316  size_t nbMembers = size_t(nbSeeds);
317  size_t nbPatterns = patterns.GetRows();
318 
320  // Setup //
322 
323  members.clear();
324 
325  double min_value = patterns.GetMin();
326  double max_value = patterns.GetMax();
327  double delta = (max_value - min_value) / double(nbSeeds);
328  double seed = min_value + delta / 2.0;
329 
330  // Heuristic setup : gaussians are placed equidistant from Min to Max value
331  // variance set to catch at least one data value
332 
333  for (size_t k = 0; k < nbMembers; ++k)
334  {
335  // Warning : if data distribution is too sparse, variance should be adapted
336 
337  bool redo = true;
338  double var = delta;
339  double sigma = sqrt(var);
340  size_t counter = 0;
341 
342  while (redo)
343  {
344  double v = patterns[counter][0];
345 
346  redo = (fabs(seed - v) > sigma);
347  ++counter;
348 
349  if ((counter >= nbPatterns) && redo)
350  {
351  var += delta;
352  sigma = sqrt(var);
353  counter = 0;
354  }
355  }
356 
357  AddMember(UnivariateGaussianPDF(seed, var), 1.0);
358 
359  seed += delta;
360  }
361 
362  MatrixDouble Proba(nbPatterns, nbMembers, 0.0);
363 
364  bool Continue = true;
365  double likelihood = 0.0;
366 
368  // Iterations //
370 
371  while(Continue)
372  {
373  // Expectation
374 
375  for (size_t i = 0; i < nbPatterns; ++i)
376  {
377  double xi = patterns[i][0];
378  double Pi = 0.0;
379 
380  for (size_t k = 0; k < nbMembers; ++k)
381  {
382  double pik = this->ValueAt(xi, k);
383 
384  Proba[i][k] = pik;
385  Pi += pik;
386  }
387 
388  Proba.MultRow(i, 1.0 / Pi);
389  }
390 
391  // Maximization
392 
393  for (size_t k = 0; k < nbMembers; ++k)
394  {
395  UnivariateGaussianPDF Gk = members[k].first;
396  double CumulPk = 0.0;
397  double muk = Gk.GetMean();
398 
399  double mu = 0.0;
400  double v = 0.0;
401 
402  for (size_t i = 0; i < nbPatterns; ++i)
403  {
404  double pik = Proba[i][k];
405  double xi = patterns[i][0];
406 
407  CumulPk += pik;
408  mu += pik * xi;
409  v += pik * Sqr(xi - muk);
410  }
411 
412  // Numeric limit reached. Recomputation needed
413  if (std::isinf(mu) || std::isinf(v))
414  {
415  mu = 0.0;
416  v = 0.0;
417 
418  for (size_t i = 0; i < nbPatterns; ++i)
419  {
420  double pik = Proba[i][k];
421  double xi = patterns[i][0];
422 
423  mu += pik * xi / CumulPk;
424  v += pik * Sqr(xi - muk) / CumulPk;
425  }
426  }
427  else
428  {
429  mu /= CumulPk;
430  v /= CumulPk;
431  }
432 
433  members[k].second = double(CumulPk) / double(nbPatterns);
434  members[k].first = UnivariateGaussianPDF(mu, v);
435  }
436 
437  double NewLikelihood = MLLE(patterns);
438  double LikelihoodDiff = fabs(NewLikelihood - likelihood);
439 
440  likelihood = NewLikelihood;
441  nbIterations++;
442 
443  Continue = ((nbIterations < maximalIterations) && (LikelihoodDiff > epsilon));
444  }
445 
446  return nbIterations;
447 }

604 std::vector<double> UnivariateGaussianMixture::MakeRandomSample(size_t n, size_t m, bool reseed) const
605 {
606  size_t nbPDF = GetNbMembers();
607 
608  auto patterns = std::vector<double>();
609 
610  std::vector<int> pop;
611  std::vector<double> cumul_weights;
612  std::vector<double> index_sample;
613 
614  // Create uniform sample to choose indexes
615  // This should become a function !!!
616 
617  if (reseed)
618  {
619  srand((unsigned int)time(nullptr));
620  }
621 
622  for (size_t k = 0; k < n; ++k)
623  {
624  index_sample.push_back(((double)(rand())) / ((double)(RAND_MAX)));
625  }
626 
627  // Cumulative weights from the mixture
628 
629  double mass = 0.0;
630 
631  for (size_t k = 0; k < nbPDF; k++)
632  {
633  mass += GetWeight(k);
634 
635  cumul_weights.push_back(mass);
636  }
637 
638  for (size_t k = 0; k < nbPDF; ++k)
639  {
640  cumul_weights[k] /= mass;
641  pop.push_back(0);
642  }
643 
644  // Build indexes to indicate which PDF of the mixture is used to generate each pattern
645 
646  for (size_t k = 0; k < n; k++)
647  {
648  double d = index_sample[k];
649  bool continue_flag = true;
650  size_t Id = 0;
651 
652  while (continue_flag)
653  {
654  if ((cumul_weights[Id] >= d) || (Id == nbPDF - 1))
655  {
656  continue_flag = false;
657  }
658  else
659  {
660  ++Id;
661  }
662  }
663 
664  ++pop[Id];
665  }
666 
667  // Create patterns
668 
669  for (size_t p = 0; p < nbPDF; ++p)
670  {
671  int subsample_pop = pop[p];
672 
673  auto subsample = GetMember(p).MakeRandomSample(subsample_pop, m, reseed);
674 
675  for (int r = 0; r < subsample_pop; ++r)
676  {
677  patterns.push_back(subsample[r]);
678  }
679  }
680 
681  return patterns;
682 }
