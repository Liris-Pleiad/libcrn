328 bool StringUTF8::EndsWith(const StringUTF8 &s) const
329 {
330  size_t length_main = Length();
331  size_t length_suffix = s.Length();
332 
333  if (length_main <= length_suffix)
334  {
335  return false;
336  }
337  else
338  {
339  size_t pos = length_main - length_suffix;
340  size_t idx = Find(s, pos);
341 
342  return (idx == pos);

518 std::vector<StringUTF8> StringUTF8::Split(const StringUTF8 &sep) const
519 {
520  size_t n = data.length();
521  auto start = data.find_first_not_of(sep.data);
522  std::vector<StringUTF8> words;
523  while (start < n)
524  {
525  auto stop = data.find_first_of(sep.data, start);
526  if (stop > n) stop = n;
527  words.push_back(data.substr(start, stop - start));
528  start = data.find_first_not_of(sep.data, stop + 1);
529  }
530  return words;
531 }

600 StringUTF8& StringUTF8::ReplaceSuffix(const StringUTF8 &old_suffix, const StringUTF8 &new_suffix)
601 {
602  size_t old_suffix_length = old_suffix.Length();
603  if (!old_suffix_length)
604  throw ExceptionInvalidArgument(_("Null suffix to search."));
605  size_t word_length = Length();
606 
607  if (old_suffix_length <= word_length)
608  {
609  size_t pivot = word_length - old_suffix_length;
610 
611  if (EndsWith(old_suffix))
612  {
613  data.erase(pivot);
614  data.append(new_suffix.Std());
615  }
616  }
617 
618  return *this;
619 }
