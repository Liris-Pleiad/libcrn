90  template<typename T> size_t StrokesWidth(const Image<T> &img, size_t maxval = 50, size_t defaultval = 0, typename std::enable_if<std::is_arithmetic<T>::value>::type *dummy = nullptr)
91  {
92  auto strokeswidth = size_t(0);
93  auto pacc = SumType<T>(0);
94  for (strokeswidth = 1; strokeswidth < Min(maxval, img.GetWidth()); strokeswidth++)
95  {
96  auto acc = SumType<T>(0);
97  for (size_t y = 0; y < img.GetHeight(); y++)
98  {
99  auto yoffset = y * img.GetWidth();
100  auto tacc = SumType<T>(0);
101  for (size_t x = 0; x < img.GetWidth(); x++)
102  {
103  auto tx = int(x) - int(strokeswidth);
104  if (tx < 0) tx += int(img.GetWidth());
105  if (tx >= int(img.GetWidth())) tx -= int(img.GetWidth());
106  tacc += Abs(img.At(x + yoffset) - img.At(tx + yoffset));
107  }
108  acc += tacc;
109  }
110  if (pacc != 0)
111  {
112  if ((DecimalType<T>(Abs(acc - pacc)) / DecimalType<T>(pacc)) < 0.1)
113  {
114  return strokeswidth;
115  }
116  }
117  pacc = acc;
118  }
119  return defaultval;
120  }

229  template<typename T> ImageBW Niblack(const Image<T> &img, size_t halfwin, double k = 0.5, typename std::enable_if<std::is_arithmetic<T>::value>::type *dummy = nullptr)
230  {
231  auto out = ImageBW(img.GetWidth(), img.GetHeight());
232  FOREACHPIXEL(x, y, img)
233  {
234  // adjust window
235  auto x1 = (x < halfwin) ? 0 : x - halfwin;
236  auto y1 = (y < halfwin) ? 0 : y - halfwin;
237  auto x2 = x + halfwin;
238  if (x2 >= img.GetWidth()) x2 = img.GetWidth() - 1;
239  auto y2 = y + halfwin;
240  if (y2 >= img.GetHeight()) y2 = img.GetHeight() - 1;
241 
242  // compute stats
243  auto sum = DecimalType<T>(0.0);
244  auto sqrsum = DecimalType<T>(0.0);
245  for (auto ty = y1; ty <= y2; ty++)
246  for (auto tx = x1; tx <= x2; tx++)
247  {
248  auto p = DecimalType<T>(img.At(tx, ty));
249  sum += p;
250  sqrsum += Sqr(p);
251  }
252  auto size = (x2 - x1 + 1) * (y2 - y1 + 1);
253  auto m = sum / double(size); // mean
254  auto sd = sqrt(sqrsum / double(size) - Sqr(m)); // std deviation
255 
256  // compute threshold
257  auto t = m + k * sd ; // original formula
258 
259  out.At(x, y) = (img.At(x, y) < T(t)) ? pixel::BWBlack : pixel::BWWhite;
260  }
261  return out;
262  }

334  template<typename T> ImageBW kMeansHisto(const Image<T> &img, size_t classes, size_t black_classes, typename std::enable_if<std::is_arithmetic<T>::value>::type *dummy = nullptr)
335  {
336  if (classes < 2 || black_classes < 1)
337  throw ExceptionDomain{"kMeansHisto(): invalid number of classes."};
338 
339  auto histo = MakeHistogram(img);
340  if (histo.Size() < 3)
341  return Fisher(img);
342 
343  auto protos = std::vector<int>(classes);
344  for (auto tmp : Range(protos))
345  {
346  protos[tmp] = int(tmp * (histo.Size() - 1) / (classes));
347  }
348  auto ok = false;
349  while (!ok)
350  {
351  auto cumul = std::vector<int>(classes, 0);
352  auto cnt = std::vector<int>(classes, 0);
353  // for each bin
354  for (auto tmp : Range(histo))
355  {
356  // find nearest prototype
357  auto c = 0;
358  auto d = Abs(int(tmp) - protos[0]);
359  for (auto tmpc = size_t(1); tmpc < classes; ++tmpc)
360  {
361  const auto td = Abs(int(tmp) - protos[tmpc]);
362  if (td < d)
363  {
364  d = td;
365  c = tmpc;
366  }
367  }
368  cumul[c] += int(tmp);
369  cnt[c] += int(histo[tmp]);
370  }
371  for (auto tmp : Range(cumul))
372  if (cnt[tmp])
373  cumul[tmp] /= cnt[tmp];
374  ok = (protos == cumul);
375  cumul.swap(protos);
376  }
377 
378  ImageBW out = ImageBW(img.GetWidth(), img.GetHeight(), pixel::BWBlack);
379  double threshold = double(protos[black_classes - 1] + protos[black_classes]) / 2.0;
380 
381  for (auto tmp : Range(img))
382  {
383  if (double(img.At(tmp)) > threshold)
384  out.At(tmp) = pixel::BWWhite;
385  else
386  out.At(tmp) = pixel::BWBlack;
387  }
388 
389  return out;
390  }

407  {

581  template<typename T> ImageBW Otsu(const Image<T> &img, typename std::enable_if<std::is_arithmetic<T>::value>::type *dummy = nullptr)
582  {
583  auto h = MakeHistogram(img);
584 
585  const auto hsize = h.Size();
586  auto mu = std::vector<double>(hsize, 0.0);
587  auto w = std::vector<double>(hsize, 0.0);
588  auto s2 = std::vector<double>(hsize, 0.0);
589  auto hnorm = std::vector<double>(hsize, 0.0);
590 
591  const auto cumul = double(h.CumulateBins());
592  for (size_t k = 0; k < hsize; ++k)
593  {
594  hnorm[k] = double(h.GetBin(k))/ cumul;
595  }
596 
597  for (size_t k = 0; k < hsize; ++k)
598  {
599  for (size_t j = 0; j < k; ++j)
600  {
601  mu[k] += double(j) * hnorm[j];
602  w[k] += hnorm[j];
603  }
604  }
605  for (size_t k = 0; k < hsize; ++k)
606  {
607  s2[k] = w[k] * (1 - w[k]) * (mu[hsize-1] * w[k] - mu[k]) * (mu[hsize-1] * w[k] - mu[k]);
608  }
609 
610  auto thresh = T{};
611  auto maxs2 = 0.0;
612  for (size_t k = 0;k < hsize; k++)
613  {
614  if (s2[k] > maxs2)
615  {
616  maxs2 = s2[k];
617  thresh = T(k);
618  }
619  }
620 
621  return Threshold(img, thresh);
622  }
