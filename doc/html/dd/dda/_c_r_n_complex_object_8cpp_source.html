281 void Savable::deserialize_internal_data(xml::Element &el)
282 {
283  // read name if any
284  const StringUTF8 nam = el.GetAttribute<StringUTF8>("name");
285 
286  xml::Element udel(el.GetFirstChildElement("Map"));
287  while (udel)
288  { // look for a map containing user data
289  const StringUTF8 role = udel.GetAttribute<StringUTF8>("role");
290  // check role
291  if (role == USERDATA_NAME)
292  { // found the user data
293  if (!user_data)
294  { // create the object if needed
295  user_data.reset(new Map());
296  }
297  else
298  { // clear the current user data if needed
299  user_data->Clear();
300  }
301  // read the user data and quit the method
302  user_data->Deserialize(udel);
303  return;
304  }
305  udel = udel.GetNextSiblingElement("Map");
306  }
307 }
