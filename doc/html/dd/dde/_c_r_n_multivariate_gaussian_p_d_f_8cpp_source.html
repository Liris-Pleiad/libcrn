101 double MultivariateGaussianPDF::ValueAt(const MatrixDouble &x) const
102 {
103  if ((x.GetCols() != 1) || (x.GetRows() != dimension))
104  throw ExceptionDimension(StringUTF8("MultivariateGaussianPDF::ValueAt(const MatrixDouble& x): ") +
105  _("incompatible dimensions"));
106 
107  // Centered input data
108  std::vector<double> y(dimension);
109 
110  for (size_t k = 0; k < dimension; ++k)
111  y[k] = x[k][0] - mean[k][0];
112 
113  // Exponential computation
114  double expo = 0.0;
115 
116  for (size_t r = 0; r < dimension; ++r)
117  {
118  double tmp = 0.0;
119 
120  for (size_t c = 0; c < dimension; ++c)
121  tmp += inverse_variance[r][c] * y[c];
122 
123  expo += tmp * y[r];
124  }
125 
126  expo /= -2.0;
127 
128  return scale_factor * exp(expo);
129 }

173 void MultivariateGaussianPDF::updateAuxiliaryAttributes()
174 {
175  scale_factor = 1.0 / (pow(2.0 * M_PI, double(dimension) / 2.0) * sqrt(variance.Determinant()));
176  inverse_variance = variance.MakeInverse();

201 bool MultivariateGaussianPDF::IsValid() const
202 {
203  if (dimension <= 1.0)
204  return false;
205 
206  for (size_t r = 0; r < dimension; ++r)
207  {
208  double mk = mean[r][0];
209 
210  if (!(mk == mk))
211  return false;
212  }
213 
214  for (size_t r = 0; r < dimension; ++r)
215  for (size_t c = 0; c < dimension; ++c)
216  {
217  double vrc = variance[r][c];
218 
219  if (!(vrc == vrc))
220  return false;
221  }
222 
223  return true;
224 }
