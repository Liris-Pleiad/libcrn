57 Path ConfigurationFile::Load()
58 {
59  std::vector<Path> dirs, files;
60  char cwd[4096];
61  getcwd(cwd, 4096);
62  dirs.push_back(cwd);
63  dirs.push_back(GetUserDirectory());
64  dirs.push_back(CRN_CONFIG_FULL_PATH);
65  files.push_back(filename + ".xml");
66  files.push_back("_" + filename + ".xml");
67  files.push_back("." + filename + ".xml");
68 
69  data.Clear();
70  for (auto idir = dirs.begin(); (idir != dirs.end()); ++idir)
71  {
72  for (auto ifn = files.begin(); (ifn != files.end()); ++ifn)
73  {
74  Path confname(*idir);
75  confname += Path::Separator();
76  confname += *ifn;
77  CRNdout << confname.CStr() << std::endl;
78  try
79  {
80  data.Load(confname);
81  CRNdout << "Configuration loaded from: " << confname.CStr() << std::endl;
82  return confname;
83  }
84  catch (...)
85  { }
86  }
87  }
88  return "";
89 }
