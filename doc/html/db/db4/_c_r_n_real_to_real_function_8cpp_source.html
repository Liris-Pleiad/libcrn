127 double RealToRealFunction::Evaluate(double x) const
128 {
129  double y = 0.0;
130 
131  if ((x >= left_endpoint) || (x <= right_endpoint))
132  {
133  size_t nbBins = GetImageSize();
134  double step = double(right_endpoint - left_endpoint) / double(nbBins - 1);
135 
136  int k = (int)((x - left_endpoint) / step);
137 
138  double x_l = left_endpoint + k * step;
139  double x_r = x_l + step;
140  double y_l = At(k);
141  double y_r = At(k + 1);
142 
143  y = ((x - x_l) * (y_r - y_l)) / (x_r - x_l) + y_l;
144  }
145 
146  return y;
147 }

177 
178  }
179

181  {

183  }
184

186 
187  double step = double(right_endpoint - left_endpoint) / double(n - 1);
188  double x_l = left_endpoint;
189  double x_r = left_endpoint;
190  double y_r = At(0);
191

196  {
