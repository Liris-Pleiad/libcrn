74 static std::pair<UImage, String> load_libpng(const Path &filename)
75 {
76  // libpng does not support URIs
77  auto fname = filename;
78  fname.ToLocal();
79 
80  /* open image file */
81  std::unique_ptr<FILE, decltype(fclose_if_not_null)*> fp(fopen(fname.CStr(), "rb"), fclose_if_not_null);
82  if (!fp)
83  return std::make_pair(UImage{}, String(_("Cannot open file ")) + fname);
84 
85  /* read magic number */
86  png_byte magic[8];
87  size_t readsize = fread(magic, 1, sizeof(magic), fp.get());
88  if (!readsize)
89  return std::make_pair(UImage{}, String(U"Empty file."));
90 
91  /* check for valid magic number */
92  if (png_sig_cmp(magic, 0, sizeof(magic)))
93  return std::make_pair(UImage{}, String(U"Not a PNG file."));
94 
95 
96  /* create a png read struct */
97  png_structp png_ptr = png_create_read_struct(PNG_LIBPNG_VER_STRING, NULL, NULL, NULL);
98  if (!png_ptr)
99  return std::make_pair(UImage{}, String(_("Cannot create PNG reader.")));
100 
101  /* create a png info struct */
102  png_infop info_ptr = png_create_info_struct(png_ptr);
103  if (!info_ptr)
104  {
105  png_destroy_read_struct(&png_ptr, NULL, NULL);
106  return std::make_pair(UImage{}, String(_("Cannot create PNG info.")));
107  }
108 
109  /* Set up the error handling */
110  if (setjmp(png_jmpbuf(png_ptr)))
111  {
112  png_destroy_read_struct(&png_ptr, &info_ptr, NULL);
113  return std::make_pair(UImage{}, String(_("Error while reading the PNG file ")) + fname);
114  }
115 
116  /* setup libpng for using standard C fread() function with our FILE pointer */
117  png_init_io(png_ptr, fp.get());
118  /* tell libpng that we have already read the magic number */
119  png_set_sig_bytes(png_ptr, sizeof(magic));
120 
121  /* Read informations */
122  png_read_info(png_ptr, info_ptr);
123  png_uint_32 width, height;
124  int bit_depth, color_type, interlace_type, compression_type, filter_method;
125  bool modified = false;
126  png_get_IHDR(png_ptr, info_ptr, &width, &height, &bit_depth, &color_type, &interlace_type, &compression_type, &filter_method);
127 
128  ColorType read_type(RGB);
129 
130  auto col0 = pixel::BWBlack;
131  auto col1 = pixel::BWWhite;
132  if (bit_depth == 1)
133  { // binary image
134  png_set_packing(png_ptr);
135  modified = true;
136  read_type = BW;
137  if (png_get_valid(png_ptr, info_ptr, PNG_INFO_PLTE))
138  { // has a palette -_-
139  // we convert it to black & white
140  png_colorp pal;
141  int numpal;
142  png_get_PLTE(png_ptr, info_ptr, &pal, &numpal);
143  if (numpal >= 2)
144  {
145  int m0 = pal[0].red + pal[0].green + pal[0].blue;
146  int m1 = pal[1].red + pal[1].green + pal[1].blue;
147  if (m1 < m0) // color 1 is darker than color 0
148  {
149  col0 = pixel::BWWhite;
150  col1 = pixel::BWBlack;
151  }
152  }
153  }
154  }
155  else if (color_type == PNG_COLOR_TYPE_GRAY || color_type == PNG_COLOR_TYPE_GRAY_ALPHA)
156  { // Grayscale image
157  if (bit_depth < 8)
158  {
159  png_set_expand_gray_1_2_4_to_8(png_ptr);
160  modified = true;
161  }
162  read_type = GRAY;
163  }
164  else if (color_type == PNG_COLOR_TYPE_PALETTE)
165  { // Changes palletted image to RGB
166  png_set_palette_to_rgb(png_ptr); // Just an alias of png_set_expand()
167  modified = true;
168  }
169  // Transform 16 bits samples to 8 bits samples
170  if (bit_depth == 16)
171  {
172  png_set_strip_16(png_ptr);
173  modified = true;
174  }
175  // Remove alpha channel
176  if (color_type & PNG_COLOR_MASK_ALPHA)
177  {
178  png_set_strip_alpha(png_ptr);
179  modified = true;
180  }
181 
182  // If any of the above were applied,
183  if (modified)
184  {
185  // After setting the transformations, libpng can update your png_info structure to reflect any transformations
186  // you've requested with this call. This is most useful to update the info structure's rowbytes field so you can
187  // use it to allocate your image memory. This function will also update your palette with the correct screen_gamma
188  // and background if these have been given with the calls above.
189  png_read_update_info(png_ptr, info_ptr);
190 
191  // Just in case any of these have changed
192  png_get_IHDR(png_ptr, info_ptr, &width, &height, &bit_depth, &color_type, &interlace_type, NULL, NULL);
193  }
194 
195  int channels = png_get_channels(png_ptr, info_ptr);
196 
197  png_bytepp row_pointers;
198 
199  /* we can now allocate memory for storing pixel data */
200  row_pointers = (png_bytep*)malloc(sizeof(png_byte*) * height);
201  for (size_t i = 0; i < height ; ++i)
202  {
203 #if (PNG_LIBPNG_VER > 10300)
204  row_pointers[i] = (png_bytep)malloc(png_get_rowbytes(png_ptr, info_ptr));
205 #else
206  row_pointers[i] = (png_bytep)malloc(info_ptr->rowbytes);
207 #endif
208  }
209  /* Read PNG file */
210  png_read_image(png_ptr, row_pointers);
211 
212  /* Conversion */
213  UImage img;
214  switch (read_type)
215  {
216  case RGB:
217  {
218  auto irgb = std::make_unique<ImageRGB>(size_t(width), size_t(height));
219  for (size_t y = 0; y < height; y++)
220  for (size_t x = 0; x < width; x++)
221  {
222  png_byte *pixel = &((row_pointers[y])[x*channels]);
223  irgb->At(x, y) = {pixel[0], pixel[1], pixel[2]};
224  }
225  img = std::move(irgb);
226  }
227  break;
228  case GRAY:
229  {
230  auto ig = std::make_unique<ImageGray>(size_t(width), size_t(height));
231  for (size_t y = 0; y < height; y++)
232  for (size_t x = 0; x < width; x++)
233  {
234  png_byte *pixel = &((row_pointers[y])[x]);
235  ig->At(x, y) = *pixel;
236  }
237  img = std::move(ig);
238  }
239  break;
240  case BW:
241  {
242  auto ibw = std::make_unique<ImageBW>(size_t(width), size_t(height));
243  for (size_t y = 0; y < height; y++)
244  for (size_t x = 0; x < width; x++)
245  {
246  png_byte *pixel = &((row_pointers[y])[x]);
247  ibw->At(x, y) = (*pixel) ? col1 : col0;
248  }
249  img = std::move(ibw);
250  }
251  break;
252  }
253 
254  /* read metadata */
255 /*
256  png_textp txt;
257  int nb = png_get_text(png_ptr, info_ptr, &txt, NULL);
258  CRNVerbose(nb);
259  for (int tmp = 0; tmp < nb; ++tmp)
260  {
261  String s(U"compression: ");
262  s += txt[tmp].compression;
263  s += U"\nkey: ";
264  s += txt[tmp].key;
265  s += U"\ntext: ";
266  s += txt[tmp].text;
267  s += U"\ntext length: ";
268  s += txt[tmp].text_length;
269 #ifdef PNG_iTXt_SUPPORTED // not supported in 1.2 branch
270  s += U"\ni text length: ";
271  s += txt[tmp].itxt_length;
272  s += U"\nlang: "
273  s += txt[tmp].lang;
274  s += U"\nlang key: "
275  s += txt[tmp].lang_key;
276 #endif
277  CRNVerbose(s);
278  }
279 */
280 
281  /* finish decompression and release memory */
282  png_read_end(png_ptr, NULL);
283  png_destroy_read_struct(&png_ptr, &info_ptr, NULL);
284  for (size_t i = 0; i < height ; ++i)
285  {
286  free(row_pointers[i]);
287  }
288  free(row_pointers);
289 
290  return std::make_pair(std::move(img), String(U""));
291 }
292 #endif

312 static std::pair<UImage, String> load_libjpeg(const Path &filename)
313 {
314  // libjpeg does not support URIs
315  auto fname(filename);
316  fname.ToLocal();
317 
318  /* open image file */
319  std::unique_ptr<FILE, decltype(fclose_if_not_null)*> fp(fopen(fname.CStr(), "rb"), fclose_if_not_null);
320  if (!fp)
321  return std::make_pair(UImage{}, String(_("Cannot open file ")) + fname);
322 
323  struct jpeg_decompress_struct cinfo;
324  struct crn_jpeg_error_mgr jerr;
325  cinfo.err = jpeg_std_error(&jerr.pub);
326  jerr.pub.error_exit = crn_jpeg_error_exit;
327  if (setjmp(jerr.setjmp_buffer))
328  {
329  jpeg_destroy_decompress(&cinfo);
330  return std::make_pair(UImage{}, String(_("Not a JPEG file.")));
331  }
332  else
333  {
334  jpeg_create_decompress(&cinfo);
335  jpeg_stdio_src(&cinfo, fp.get());
336  jpeg_read_header(&cinfo, TRUE);
337  jpeg_start_decompress(&cinfo);
338  int w = cinfo.output_width;
339  int h = cinfo.output_height;
340  int bpp = cinfo.out_color_components;
341  UImageRGB irgb;
342  UImageGray ig;
343  if (bpp == 3)
344  {
345  irgb = std::make_unique<ImageRGB>(w, h);
346  }
347  else if (bpp == 1)
348  {
349  ig = std::make_unique<ImageGray>(w, h);
350  }
351  else
352  {
353  jpeg_finish_decompress(&cinfo);
354  jpeg_destroy_decompress(&cinfo);
355  return std::make_pair(UImage{}, String(_("JPEG file contains unnatural bytes per pixel count.")));
356  }
357  JSAMPARRAY buffer = (*cinfo.mem->alloc_sarray)((j_common_ptr) &cinfo, JPOOL_IMAGE, w * bpp, 1);
358  int y = 0;
359  while (cinfo.output_scanline < cinfo.output_height)
360  {
361  (void) jpeg_read_scanlines(&cinfo, buffer, 1);
362  for (int x = 0; x < w; ++x)
363  {
364  if (bpp == 3)
365  {
366  irgb->At(x, y) = { buffer[0][3 * x], buffer[0][3 * x + 1], buffer[0][3 * x + 2] };
367  }
368  else
369  {
370  ig->At(x, y) = buffer[0][x];
371  }
372  }
373  y += 1;
374  }
375 
376  // read comments
377  // TODO
378  // jpeg_read_header() then cinfo->marker_list (linked list)
379 
380  jpeg_finish_decompress(&cinfo);
381  jpeg_destroy_decompress(&cinfo);
382  if (bpp == 3)
383  return std::make_pair(std::move(irgb), String(U""));
384  else
385  return std::make_pair(std::move(ig), String(U""));
386  }
387 }

419  }

421  g_free(filename);
422 
423  if (!pb)
424  {
425  String msg((char*)err->message);
426  g_error_free(err);
427  return std::make_pair(UImage{}, String(_("Cannot open image. ")) + msg);
428  }
429  int w = gdk_pixbuf_get_width(pb);
430  int h = gdk_pixbuf_get_height(pb);
431  int rs = gdk_pixbuf_get_rowstride(pb);
432  int word = gdk_pixbuf_get_has_alpha(pb) ? 4 : 3;
433 
434  UImageRGB img(std::make_unique<ImageRGB>(w, h));
435  guchar *oripix = gdk_pixbuf_get_pixels(pb);
436  for (int y = 0; y < h; y++)
437  for (int x = 0; x < w; x++)
438  {
439  const auto o2 = word * x + y * rs;
440  img->At(x, y) = { oripix[o2], oripix[o2 + 1], oripix[o2 + 2] };
441  }
442  g_object_unref(pb);
443  return std::make_pair(std::move(img), String(U""));
444 }
