155 UMatrixDouble DiscreteHMM::alpha(const MatrixInt& observed) const
156 {
157  size_t i, j, t;
158  size_t N = nbStates;
159  size_t T = observed.GetCols();
160 
161  UMatrixDouble FWD = std::make_unique<MatrixDouble>(T, N, 0.0);
162  // Just some convenient short variable names to fit literature and make it easier to handle ! =)
163  SSquareMatrixDouble A = stateTransitionProbability;
164  SMatrixDouble B = stateGivenSymbolProbability;
165  SMatrixDouble P = firstStateProbability;
166 
167  // First alpha for each state.
168 
169  for (i = 0; i < N; i++)
170  {
171  FWD->At(0, i) = P->At(i, 0) * B->At(i, observed.At(0, 0));
172  }
173 
174  // Iteration.
175 
176  for (t = 0; t < T - 1; t++)
177  {
178  for (j = 0; j < N; j++)
179  {
180  double Sum = 0.0;
181 
182  for (i = 0; i < N; i++)
183  {
184  Sum += FWD->At(t, i) * A->At(i, j);
185  }
186 
187  FWD->At(t + 1, j) = Sum * B->At(j, observed.At(t + 1, 0));
188  }
189  }
190 
191  return std::forward<UMatrixDouble>(FWD);
192 }

201 UMatrixDouble DiscreteHMM::beta(const MatrixInt& observed) const
202 {
203  size_t i,j;
204  size_t N = nbStates;
205  size_t T = observed.GetCols();
206 
207  UMatrixDouble BKD = std::make_unique<MatrixDouble>(T, N, 0.0);
208  SSquareMatrixDouble A = stateTransitionProbability;
209  SMatrixDouble B = stateGivenSymbolProbability;
210 
211  // Last beta for each state.
212 
213  for (i = 0; i < N; i++)
214  {
215  BKD->At(T - 1, i) = 1.0;
216  }
217 
218  // Iteration.
219 
220  for (int t = int(T) - 2; t >= 0; t--)
221  {
222  for (i = 0; i < N; i++)
223  {
224  double Sum = 0.0;
225 
226  for (j = 0; j < N; j++)
227  {
228  Sum += A->At(i, j) *
229  B->At(j, observed.At(t+1, 0)) *
230  BKD->At(t + 1, j);
231  }
232 
233  BKD->At(t, i) = Sum;
234  }
235  }
236 
237  return std::forward<UMatrixDouble>(BKD);

270 UMatrixInt DiscreteHMM::MakeViterbi(const MatrixInt& observed) const
271 {
272  size_t i, j, k;
273  size_t T = observed.GetRows();
274  size_t N = nbStates;
275 
276  SSquareMatrixDouble A = stateTransitionProbability;
277  SMatrixDouble B = stateGivenSymbolProbability;
278  SMatrixDouble P = firstStateProbability;
279 
280  SMatrixDouble delta = std::make_shared<MatrixDouble>(T, N, 0.0);
281  SMatrixInt phi = std::make_shared<MatrixInt>(T, N);
282 
283  // Setup.
284 
285  for (i = 0; i < N; i++)
286  {
287  delta->At(0, i) = P->At(i, 0) * B->At(i, observed.At(0, 0));
288  phi->At(0, i) = 0;
289  }
290 
291  // Recursion.
292 
293  for (size_t t = 1; t < T; t++)
294  {
295  // New delta values.
296 
297  double NewDelta = 0.0;
298  double MaxDelta = 0.0;
299  size_t Index = 0;
300 
301  for (j = 0; j < N; j++)
302  {
303  for (i = 0; i < N; i++)
304  {
305  NewDelta = delta->At(t - 1, i) * A->At(i, j);
306 
307  if (NewDelta > MaxDelta)
308  {
309  MaxDelta = NewDelta;
310  Index = i;
311  }
312  }
313 
314  delta->At(t, j) = MaxDelta * B->At(j, observed.At(t, 0));
315  phi->At(t, j) = int(Index);
316  }
317  }
318 
319  // Backtracking.
320 
321  UMatrixInt Path = std::make_unique<MatrixInt>(T, 1);
322  size_t CurrentState;
323 
324  SMatrixDouble LastDelta = std::make_shared<MatrixDouble>(N, 1, 0.0);
325 
326  for (k = 0; k < N; k++)
327  {
328  LastDelta->At(k, 0) = delta->At(T - 1, k);
329  }
330 
331  CurrentState = LastDelta->ArgmaxInColumn(0);
332 
333  Path->At(T - 1, 0) = int(CurrentState);
334 
335  for (int t = int(T) - 2; t >= 0; t--)
336  {
337  CurrentState = phi->At(t + 1, CurrentState);
338  Path->At(t, 0) = int(CurrentState);
339  }
340 
341  return std::forward<UMatrixInt>(Path);
342 }

350 void DiscreteHMM::BaumWelchSingle(const MatrixInt& observed, size_t maxIter)
351 {
352  size_t Count = 1;
353  bool Continue = true;
354 
355  while (Continue)
356  {
357  SDiscreteHMM H = CloneAs<DiscreteHMM>(*this);
358 
359  size_t i, j, k, t;
360  size_t N = nbStates;
361  size_t K = nbSymbols;
362  size_t T = observed.GetCols();
363 
364  SquareMatrixDouble A = *stateTransitionProbability;
365  MatrixDouble B = *stateGivenSymbolProbability;
366  MatrixDouble P = *firstStateProbability;
367 
368  UMatrixDouble Alpha(alpha(observed));
369  UMatrixDouble Beta(beta(observed));
370  SMatrixDouble Gamma = std::make_shared<MatrixDouble>(T, N, 0.0);
371 
372  std::vector<SSquareMatrixDouble> Xi(T);
373 
374  for (size_t date = 0; date < T; date++)
375  {
376  Xi[date] = std::make_shared<SquareMatrixDouble>(N, 0.0);
377  }
378 
379  // Compute sequence probability via Alpha.
380 
381  double Prob = 0.0;
382 
383  for (k = 0; k < N; k++)
384  {
385  Prob += Alpha->At(T - 1, k);
386  }
387 
389  // Computation for Xi and Gamma. //
391 
392  for (t = 0; t < T - 1; t++)
393  {
394  for (i = 0; i < N; i++)
395  {
396  for (j = 0; j < N; j++)
397  {
398  double z = (Alpha->At(t, i) *
399  A.At(i, j) *
400  B.At(j, observed.At(t + 1, 0)) *
401  Beta->At(t + 1, j)) / Prob;
402 
403  Xi[t]->At(i, j) = z;
404 
405  Gamma->IncreaseElement(t, i, z);
406  }
407  }
408  }
409 
410  // Last gamma vector (cf. Rabiner & Juang).
411 
412  for (i = 0; i < N; i++)
413  {
414  double alpha = Alpha->At(T - 1, i);
415  double beta = Beta->At(T - 1, i);
416  double g = alpha * beta / Prob;
417 
418  Gamma->IncreaseElement(T - 1, i, g);
419  }
420 
422  // Re-estimation. //
424 
425  // Re-estimation for P.
426 
427  P = Gamma->MakeRow(0);
428 
429  // Re-estimation for A.
430 
431  A.SetAll(0.0);
432 
433  SMatrixDouble InvScales = std::make_shared<MatrixDouble>(N, 1, 0.0);
434 
435  for (t = 0; t < T - 1; t++)
436  {
437  for (i = 0; i < N; i++)
438  {
439  InvScales->IncreaseElement(i, 0, Gamma->At(t, i));
440 
441  for (j = 0; j < N; j++)
442  {
443  A.IncreaseElement(i, j, Xi[t]->At(i, j));
444  }
445  }
446  }
447  Xi.clear();
448 
449  for (i = 0; i < N; i++)
450  {
451  for (j = 0; j < N; j++)
452  {
453  double v = A.At(i, j);
454 
455  A.At(i, j) = v / InvScales->At(i, 0);
456  }
457  }
458 
459  // Re-estimation for B.
460 
461  for (j = 0; j < N; j++)
462  {
463  for (k = 0; k < K; k++)
464  {
465  double Numerator = 0.0;
466  double Denominator = 0.0;
467 
468  for (t = 0; t < T; t++)
469  {
470  double g = Gamma->At(t, j);
471 
472  if (observed.At(t, 0) == int(k))
473  {
474  Numerator += g;
475  }
476 
477  Denominator += g;
478  }
479 
480  B.At(j, k) = Numerator / Denominator;
481  }
482  }
483 
484  firstStateProbability = CloneAs<MatrixDouble>(P);
485  stateTransitionProbability = CloneAs<SquareMatrixDouble>(A);
486  stateGivenSymbolProbability = CloneAs<MatrixDouble>(B);
487 
489  // End for Baum-Welch step. //
491 
492  Count++;
493  Continue = !((*this == *H) || (Count > maxIter));
494  }
495 
496  // XXX I forgot if this thing is still useful !!!
497  //this.updateCumuls();
498 
499 }

507 void DiscreteHMM::BaumWelchMultiple(const MatrixInt& observationSet, size_t maxIter)
508 {
509  size_t Count = 1;
510  bool Continue=true;
511  size_t i, j, k, m, n, t;
512  size_t N = nbStates;
513  size_t M = nbSymbols;
514  size_t nbPatterns = observationSet.GetRows();
515 
516  while (Continue)
517  {
518  SDiscreteHMM H = CloneAs<DiscreteHMM>(*this);
519 
520  SSquareMatrixDouble A = stateTransitionProbability;
521  SMatrixDouble B = stateGivenSymbolProbability;
522  SMatrixDouble P = firstStateProbability;
523 
524  SMatrixDouble CumulNumeratorA = std::make_shared<MatrixDouble>(N, N, 0.0);
525  SMatrixDouble CumulNumeratorB = std::make_shared<MatrixDouble>(N, M, 0.0);
526  SMatrixDouble CumulDenominatorA = std::make_shared<MatrixDouble>(N, N, 0.0);
527  SMatrixDouble CumulDenominatorB = std::make_shared<MatrixDouble>(N, M, 0.0);
528  SMatrixDouble CumulP = std::make_shared<MatrixDouble>(N, 1, 0.0);
529 
531  // Loop over all sequences. //
533 
534  for (k = 0; k < nbPatterns; k++)
535  {
536  MatrixInt Sample = observationSet.MakeRowAsColumn(k);
537  size_t T = Sample.GetRows();
538 
539  UMatrixDouble Alpha(alpha(Sample));
540  UMatrixDouble Beta(beta(Sample));
541 
542  double Prob = 0.0;
543 
544  for (i = 0; i < N; i++)
545  {
546  Prob += Alpha->At(T - 1, i);
547  }
548 
549  std::vector<SSquareMatrixDouble> Xi;
550  std::vector<SMatrixDouble> Gamma;
551 
552 
554  // Computation for Xi and Gamma. //
556 
557  for (t = 0; t < T - 1; t++)
558  {
559  SSquareMatrixDouble xi = std::make_shared<SquareMatrixDouble>(N);
560  SMatrixDouble gamma = std::make_shared<MatrixDouble>(N, 1, 0.0);
561 
562  for (i = 0; i < N; i++)
563  {
564  for (j = 0; j < N; j++)
565  {
566  double z = (Alpha->At(t, i)*
567  A->At(i, j)*
568  B->At(j, Sample.At(t + 1, 0))*
569  Beta->At(t + 1, j)) / Prob;
570 
571  xi->At(i, j) = z;
572  gamma->IncreaseElement(i, 0, z);
573  }
574  }
575 
576  Xi.push_back(xi);
577  Gamma.push_back(gamma);
578  }
579 
580  // Last gamma vector.
581 
582  {
583  SMatrixDouble gamma = std::make_shared<MatrixDouble>(N, 1, 0.0);
584  double Denominator = 0.0;
585 
586  for (i = 0; i < N; i++)
587  {
588  double alpha = Alpha->At(T - 1, i);
589  double beta = Beta->At(T - 1, i);
590  Denominator += alpha * beta;
591  }
592 
593  for (i = 0; i < N; i++)
594  {
595  double alpha=Alpha->At(T - 1, i);
596  double beta=Beta->At(T - 1, i);
597  gamma->At(i, 0) = (alpha * beta) / Denominator;
598  }
599 
600  Gamma.push_back(gamma);
601  }
602 
604  // Accumulation. //
606 
607  // Accumulation for A.
608 
609  for (m = 0; m < N; m++)
610  {
611  for (n = 0; n < N; n++)
612  {
613  for (t = 0; t < T - 1; t++)
614  {
615  CumulNumeratorA->IncreaseElement(m, n, Xi[t]->At(m, n));
616  CumulDenominatorA->IncreaseElement(m, n, Gamma[t]->At(m, 0));
617  }
618  }
619  }
620 
621  // Accumulation for B;
622 
623  for (n = 0; n < N; n++)
624  {
625  for (m = 0; m < M; m++)
626  {
627  for (t = 0; t < T; t++)
628  {
629  double g = Gamma[t]->At(n, 0);
630 
631  if (Sample.At(t, 0) == int(m))
632  {
633  CumulNumeratorB->IncreaseElement(n, m, g);
634  }
635 
636  CumulDenominatorB->IncreaseElement(n, m, g);
637  }
638  }
639  }
640 
641  // Accumulation for P.
642 
643  for (n = 0; n < N; n++)
644  {
645  CumulP->IncreaseElement(n, 0, Gamma[0]->At(n, 0));
646  }
647  }
648 
650  // Levinson's re-estimation. //
652 
653  for (m = 0; m < N; m++)
654  {
655  for (n = 0; n < N; n++)
656  {
657  if (A->At(m, n) != 0.0)
658  {
659  A->At(m, n) = CumulNumeratorA->At(m, n) / CumulDenominatorA->At(m, n);
660  }
661  }
662  }
663 
664  for (n = 0; n < N; n++)
665  {
666  for (m = 0; m < M; m++)
667  {
668  if (B->At(n,m) != 0.0)
669  {
670  B->At(n, m) = CumulNumeratorB->At(n, m) / CumulDenominatorB->At(n, m);
671  }
672  }
673  }
674 
675  *CumulP *= 1.0 / double(nbSymbols);
676 
677  for (n = 0; n < N; n++)
678  {
679  P->At(n, 0) = CumulP->At(n, 0);
680  }
681 
683  // Maintain consistency. //
685 
686  //forceConsistency();
687 
689  // End for Baum-Welch step. //
691 
692  Count++;
693 
694  Continue = !((*this == *H) || (Count > maxIter));
695  }
696

698  /*
699  this.updateCumuls();
700  this.firstStateProbabilityCumul.display();
701  this.stateTransitionProbabilityCumul.display();
702  this.stateGivenSymbolProbabilityCumul.display();

709 void DiscreteHMM::forceConsistency()
710 {
711  size_t N = nbStates;
712  size_t K = nbSymbols;
713  SSquareMatrixDouble A = stateTransitionProbability;
714  SMatrixDouble B = stateGivenSymbolProbability;
715 
716  // Sum over each row in A should be 1.
717 
718  for (size_t l = 0; l < N; l++)
719  {
720  double Cumul=0.0;
721 
722  for (size_t c = 0; c < N; c++)
723  {
724  Cumul += A->At(l, c);
725  }
726 
727  if (Cumul != 1.0)
728  {
729  for (size_t c = 0; c < N; c++)
730  {
731  double a = A->At(l, c)/Cumul;
732  A->At(l, c) = a;
733  }
734  }
735  }
736 
737  // Sum over row line in B should be 1.
738 
739  for (size_t l = 0; l < N; l++)

741  double Cumul = 0.0;
742 
743  for (size_t c = 0; c < K; c++)
744  {
745  Cumul += B->At(l, c);
746  }
747 
748  if (Cumul != 1.0)
