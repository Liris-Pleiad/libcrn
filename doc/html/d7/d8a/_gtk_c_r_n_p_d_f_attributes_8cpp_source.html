143 Glib::ustring PDFAttributes::layout2str(crn::PDF::Attributes::Layout lay)
144 {
145  switch (lay)
146  {
147  case crn::PDF::Attributes::Layout::ONE_PAGE:
148  return _(onePage.c_str());
149  case crn::PDF::Attributes::Layout::CONTINUOUS:
150  return _(continuous.c_str());
151  case crn::PDF::Attributes::Layout::OPEN_LEFT:
152  return _(openLeft.c_str());
153  case crn::PDF::Attributes::Layout::OPEN_RIGHT:
154  return _(openRight.c_str());
155  default:
156  return _(onePage.c_str());
157  }
158 }

160 crn::PDF::Attributes::Layout PDFAttributes::str2layout(const Glib::ustring &lay)
161 {
162  if (lay == _(onePage.c_str()))
163  return crn::PDF::Attributes::Layout::ONE_PAGE;
164  if (lay == _(continuous.c_str()))
165  return crn::PDF::Attributes::Layout::CONTINUOUS;
166  if (lay == _(openLeft.c_str()))
167  return crn::PDF::Attributes::Layout::OPEN_LEFT;
168  if (lay == _(openRight.c_str()))
169  return crn::PDF::Attributes::Layout::OPEN_RIGHT;
170  return crn::PDF::Attributes::Layout::ONE_PAGE;
171 }

308 void PDFAttributes::on_owner_pass_changed()
309 {
310  attr.owner_password = owner_pass.get_text().c_str();
311  if (attr.owner_password.IsEmpty())
312  {
313  attr.user_password = "";
314  attr.copyable = true;
315  attr.printable = true;
316  user_pass.set_sensitive(false);
317  copyable.set_sensitive(false);
318  printable.set_sensitive(false);
319  }
320  else
321  {
322  attr.user_password = user_pass.get_text().c_str();
323  attr.copyable = copyable.get_active();
324  attr.printable = printable.get_active();
325  user_pass.set_sensitive(true);
326  copyable.set_sensitive(true);
327  printable.set_sensitive(true);
328  }
329 }
