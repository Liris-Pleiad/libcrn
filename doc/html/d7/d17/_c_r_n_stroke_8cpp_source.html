65 {

123 {

153 void Stroke::MergeWithXInterpolation(const Stroke &str, unsigned int step)
154 {
155  if (&str == this)
156  throw ExceptionInvalidArgument(_("Invalid stroke"));
157 
158  if (step <= 0)
159  {
160  throw ExceptionDomain(StringUTF8("void Stroke::MergeWithXInterpolation("
161  "const Stroke &str, double step): ") +
162  _("Null step."));
163  }
164  auto p = points.back();
165  int bx = p.X;
166  double by = p.Y;
167  p = str.points.front();
168  int ex = p.X;
169  double ey = p.Y;
170 
171  if (Abs(ex - bx) > int(step))
172  {
173  double yoffset = (ey - by) / Abs(ex - bx);
174  double y = by + yoffset;
175  if (bx > ex)
176  { // right to left
177  for (int x = bx - step; x > ex; x -= step)
178  {
179  AddPoint(Point2DInt(x, (int)y));
180  y += yoffset;
181  }
182  }
183  else
184  { // left to right
185  for (int x = bx + step; x < ex; x += step)
186  {
187  AddPoint(Point2DInt(x, (int)y));
188  y += yoffset;
189  }
190  }
191  }
192 
193  for (const auto &p : str.points)
194  {
195  AddPoint(p);
196  }
197  bbox = nullptr;
198 }

258 Rect Stroke::GetBBox() const
259 {
260  if (points.size() > 0)
261  {
262  // is the bbox already computed?
263  if (bbox)
264  return *bbox;
265  // else update bbox
266  auto fp = points.front();
267  Rect r(fp.X, fp.Y, fp.X, fp.Y);
268  for (size_t tmp = 1; tmp < points.size(); tmp++)
269  {
270  const auto &p = points[tmp];
271  r |= Rect(p.X, p.Y, p.X, p.Y);
272  }
273  bbox = CloneAs<Rect>(r); // store to prevent recomputing
274  return r;
275  }
276  else
277  return Rect();
278 }
