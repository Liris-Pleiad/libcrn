582  // width scaling
583  auto wscale = std::vector<pixel_type>();

585

587  { // just copy (assuming the values are base types)
588  wscale.swap(pixels);
589  }

591  {
592  wscale = std::vector<pixel_type>(w * height, defaultvalue);
593  for (auto x = size_t(0); x < w; ++x)
594  {
595  auto min = int(double(x) * step);
596  auto coeffmin = 1 - (double(x) * step - double(min));
597 
598  auto max = int(double(x + 1) * step);
599  auto coeffmax = double(x + 1) * step - double(max);
600

602  {

611  auto coeff = 0.0;

613

615  coeff += coeffmin;
616  if (coeffmax)
617  {
618  acc += coeffmax * pixels[max + yoff];
619  coeff += coeffmax;
620  }
621  for (int k = min + 1; k < max; ++k)
622  {
623  acc += pixels[k + yoff];
624  coeff += 1;
625  }

630  }
