354 void Document::ReorderViewsFrom(const std::vector<size_t> &from)
355 {
356  if (from.size() != views.size())
357  {
358  throw ExceptionDimension(StringUTF8("bool Document::ReorderViewsFrom(const std::vector<size_t> &from): ") +
359  _("bad changeset size."));
360  }
361  std::set<size_t> cont;
362  for (size_t i : from)
363  {
364  cont.insert(i);
365  }
366  if (cont.size() != views.size())
367  {
368  throw ExceptionLogic(StringUTF8("bool Document::ReorderViewsFrom(const std::vector<size_t> &from): ") +
369  _("changeset contains duplicates."));
370  }
371  if ((*cont.begin() != 0) || (*cont.rbegin() != views.size() - 1))
372  {
373  throw ExceptionDomain(StringUTF8("bool Document::ReorderViewsFrom(const std::vector<size_t> &from): ") +
374  _("changeset contains values out of bounds."));
375  }
376  std::vector<view> newviews;
377  for (size_t i : from)
378  {
379  newviews.push_back(views[i]);
380  }
381  views.swap(newviews);
382 }

593 Path Document::GetThumbnailFilename(size_t index, bool refresh) const
594 {

596  throw ExceptionDomain(StringUTF8("const Path Document::GetThumbnailFilename(size_t index, bool refresh) const: ") + _("index out of bounds."));

643 void Document::save(const Path &fname)
644 {
645  size_t namepos = fname.BackwardFindAnyOf("."); // strip extension if any
646  if (namepos == String::NPos())
647  namepos = 0;
648  basename = Path(fname.SubString(0, namepos)) + "_data"; // append "_data"
649  if (!IO::Access(basename, IO::EXISTS))
650  {
651  IO::Mkdir(basename); // may throw
652  }
653 
654  xml::Document doc;
655  doc.PushBackComment("libcrn Document file");
656  xml::Element root(doc.PushBackElement("Document"));
657  root.SetAttribute("basename", basename.CStr());
658  root.SetAttribute("author", author.CStr());
659  root.SetAttribute("date", date.CStr());
660 
661  // save views
662  for (size_t tmp = 0; tmp < views.size(); tmp++)
663  {
664  xml::Element el(root.PushBackElement("View"));
665  el.SetAttribute("fname", views[tmp].filename.CStr());
666  el.SetAttribute("id", views[tmp].id.CStr());
667  el.SetAttribute("num", int(tmp));
668  }
669 
670  serialize_internal_data(root);
671 
672  doc.Save(fname); // may throw
673 }

751 void Document::ExportPDF(const Path &fname, const PDF::Attributes &attr, Progress *prog) const
752 {
753  PDF::Doc pdf(attr);
754  std::vector<Path> images;
755  Path tmpimg(tmpnam(nullptr));
756  if (prog)
757  prog->SetMaxCount((int)GetNbViews());
758  for (const view &v : views)
759  {
760  PDF::Page page = pdf.AddPage();
761  PDF::Image image;
762  if (attr.lossy_compression)
763  { // jpeg
764  auto img = NewImageFromFile(v.filename);
765  img->SaveJPEG(tmpimg, attr.jpeg_qual);
766  image = pdf.AddJPEG(tmpimg);
767  }
768  else
769  { // png
770  auto img = NewImageFromFile(v.filename);
771  img->SavePNG(images.back());
772  images.push_back(tmpnam(nullptr));
773  image = pdf.AddPNG(images.back());
774  }
775  page.SetWidth((double)image.GetWidth());
776  page.SetHeight((double)image.GetHeight());
777  page.DrawImage(image, {0, 0, (int)image.GetWidth() - 1, (int)image.GetHeight() - 1});
778  if (prog)
779  prog->Advance();
780  }
781  // save
782  pdf.Save(fname);
783 
784  if (attr.lossy_compression)
785  {
786  try
787  {
788  IO::Rm(tmpimg);
789  } catch (...) { }
790  }
791  else
792  {
793  for (const Path &fname : images)
794  {
795  try
796  {
797  IO::Rm(fname);
798  } catch (...) { }
