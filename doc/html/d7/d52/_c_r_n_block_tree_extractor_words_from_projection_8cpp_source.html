41 void BlockTreeExtractorWordsFromProjection::Extract(Block &b)
42 {
43  if (!b.HasTree(connectedComponentTreeName))
44  {
45  // Check if block contains a tree of connected components
46  // If not, perform a component extraction
47  b.ExtractCC(connectedComponentTreeName);
48 
49  // Drop connected components with width AND height smaller than 2
50  b.FilterMinAnd(connectedComponentTreeName, 2, 2);
51  }
52 
53  size_t nbCCs = b.GetNbChildren(connectedComponentTreeName);
54 
55  if (nbCCs == 0)
56  {
57  // No connected component in this block => no word !!!
58  // Nothing to be done but no error. Just return.
59  return;
60  }
61 
62  /*** General case ***/
63 
64  // Step 1 : vertical projection
65 
66  auto vProj = VerticalProjection(*b.GetBW());
67 
68  // Step 2 : find white streams (start points, end points, lengths)
69  // in profile
70 
71  std::vector< size_t > whiteStreamStartPoints;
72  std::vector< size_t > whiteStreamEndPoints;
73  std::vector< double > whiteStreamLengths;
74 
75  bool inWhiteStream = (vProj.GetBin(0) == 0);
76  size_t n = vProj.Size();
77  double l = 0.0;
78 
79  if (inWhiteStream)
80  {
81  l = 1.0;
82  }
83 
84  for (size_t k = 1; k < n; k++)
85  {
86  if (vProj.GetBin(k) == 0)
87  {
88  // Case 1 : void bin
89 
90  if (!inWhiteStream)
91  {
92  // Case 1.1 : begin a new white stream
93  inWhiteStream = true;
94  whiteStreamStartPoints.push_back(k);
95  l = 1.0;
96  }
97  else
98  {
99  // Case 1.2 : continue white stream
100  l += 1.0;
101  }
102  }
103  else
104  {
105  // Case 2 : non-void bin
106 
107  if (inWhiteStream)
108  {
109  // Case 2.1 : run out of a white stream
110  inWhiteStream = false;
111  whiteStreamEndPoints.push_back(k - 1);
112  whiteStreamLengths.push_back(l);
113  }
114 
115  }
116  }
117 
118  // Step 3 : statistical calculus
119 
120  // Pattern matrix to store white stream lengths
121  size_t nbWhiteStreams = whiteStreamLengths.size();
122  SMatrixDouble data = std::make_shared<MatrixDouble>(nbWhiteStreams, 1, 0.0);
123 
124  for (size_t k = 0; k < nbWhiteStreams; k++)
125  {
126  data->At(k, 0) = whiteStreamLengths[k];
127  }
128 
129  double mu = data->MakeColumnMeans().At(0, 0);
130  double rho = data->MakeCovariance().At(0, 0);
131 
132  // Create univariate gaussian mixture
133  SUnivariateGaussianMixture ugm = std::make_shared<UnivariateGaussianMixture>();
134 
135  // Try EM optimisation and compute likelihood with 1 PDF hypothesis
136  ugm->EM(*data, 1);
137  double mlle1 = ugm->MLLE(*data);
138  // Try EM optimisation and compute likelihood with 2 PDF hypothesis
139  ugm->EM(*data, 2);
140  double mlle2 = ugm->MLLE(*data);
141 
142  // Get high and low values means of PDFs in mixture
143  double m_low = ugm->GetMember(0).GetMean();
144  double m_high = ugm->GetMember(1).GetMean();
145 
146  // Check and swap values if needed
147  if (m_low > m_high)
148  {
149  double tmp = m_high;
150 
151  m_high = m_low;
152  m_low = tmp;
153  }
154 
155  // Step 4 : cases for box splitting
156 
157  int nb_words = 1;
158  double avg_between_word_spacing = 0;
159 
160  Rect bb = b.GetAbsoluteBBox();
161  int leftSide = bb.GetLeft();
162  int rightSide = bb.GetRight();
163  int topSide = bb.GetTop();
164  int bottomSide = bb.GetBottom();
165 
166  if (mlle2 == std::numeric_limits<double>::infinity())
167  {
168  // Case 1 : one member has null variance in the 2 PDF mixture
169  // i.e. the PDF was generated by a single data
170 
171  if (rho == 0.0)
172  {
173  // Case 1.1 : all values equal in data set
174  // No need to spit. Block simply added as is.
175  b.AddChildAbsolute(GetTreeName(), b.GetAbsoluteBBox());
176  }
177  else
178  {
179  // Case 1.2 : one outlier value in data set
180 
181  double outlier;
182 
183  if (ugm->GetMember(0).GetVariance() == 0.0)
184  {
185  outlier = ugm->GetMember(0).GetMean();
186  }
187  else
188  {
189  outlier = ugm->GetMember(1).GetMean();
190  }
191 
192  if (outlier < mu)
193  {
194  // Case 1.2.1 : outlier is smaller than average gap lengths
195  // No need to spit. Block simply added as is.
196  b.AddChildAbsolute(GetTreeName(), b.GetAbsoluteBBox());
197  }
198  else
199  {
200  // Case 1.2.2 : outlier is greater than average gap lengths
201  // Split in two sub-blocks
202  int k = 0;
203 
204  while (whiteStreamLengths[k] != outlier)
205  {
206  k++;
207  }
208 
209  b.AddChildAbsolute(GetTreeName(), Rect(leftSide, topSide, leftSide + int(whiteStreamStartPoints[k]) - 1, bottomSide));
210  b.AddChildAbsolute(GetTreeName(), Rect(leftSide + int(whiteStreamEndPoints[k]) + 1, topSide, rightSide, bottomSide));
211 
212  avg_between_word_spacing = outlier;
213  nb_words = 2;
214  }
215  }
216  }
217  else
218  {
219  // Case 2 : no member has null variance in the 2 PDF mixture
220 
221  if (mlle2 > mlle1)
222  {
223  // Case 2.1 : data set likely comes from two distributions
224  // Catch high values in data set and split block at
225  // gaps corresponding to large white streams
226 
227  // Check white streams and keep the long ones
228  std::vector< size_t > gapStartPoints;
229  std::vector< size_t > gapEndPoints;
230  std::vector< double > gapWidths;
231 
232  for (size_t k = 0; k < nbWhiteStreams; k++)
233  {
234  double lk = whiteStreamLengths[k];
235 
236  if (fabs(lk - m_high) < fabs(lk - m_low))
237  {
238  gapStartPoints.push_back(whiteStreamStartPoints[k]);
239  gapEndPoints.push_back(whiteStreamEndPoints[k]);
240  gapWidths.push_back(whiteStreamLengths[k]);
241  }
242  }
243 
244  size_t nbGaps = gapStartPoints.size();
245 
246  if (gapStartPoints[0] > 0)
247  {
248  b.AddChildAbsolute(GetTreeName(), Rect(leftSide, topSide, leftSide + int(gapStartPoints[0]) - 1, bottomSide));
249  }
250 
251  for (size_t k = 1; k < nbGaps; k++)
252  {
253  b.AddChildAbsolute(GetTreeName(), Rect(leftSide + int(gapEndPoints[k - 1]) + 1, topSide, leftSide + int(gapStartPoints[k]) - 1, bottomSide));
254  avg_between_word_spacing += gapWidths[k];
255  nb_words++;
256  }
257 
258  if (int(gapEndPoints[nbGaps - 1]) < rightSide)
259  {
260  b.AddChildAbsolute(GetTreeName(), Rect(leftSide + int(gapEndPoints[nbGaps - 1]) + 1, topSide, rightSide, bottomSide));
261  avg_between_word_spacing += gapWidths[nbGaps - 1];
262  nb_words++;
263  }
264  }
265  else
266  {
267  // Case 2.2 : data set likely comes from a unique distributions
268  // No need to spit. Block simply added as is.
269  b.AddChildAbsolute(GetTreeName(), b.GetAbsoluteBBox());
270  }
271 
272  }
273 
274  avg_between_word_spacing /= nb_words;
275 
276  b.SetUserData(U"nb_words", std::make_shared<Int>(nb_words));
277  b.SetUserData(U"avg_between_word_spacing", std::make_shared<Int>((int)(avg_between_word_spacing)));
278 }
