42 PCA::PCA(const MatrixDouble &data, bool data_reduction_flag)
43 {
44  dimension = data.GetCols();
45 
46  size_t nb_patterns = data.GetRows();
47  bool status = ((nb_patterns > 1) && (dimension > 1));
48 
49  if (status)
50  {
51  means = std::vector<double>(dimension, 0.0);
52  deviations = std::vector<double>(dimension, 0.0);
53 
54  // Simultaneous computation of means and
55  // means of squared data (useful for variance further computation)
56  std::vector<double> means_of_squares(dimension, 0.0);
57 
58  for (size_t k = 0; k < nb_patterns; ++k)
59  for (size_t d = 0; d < dimension; ++d)
60  {
61  double val = data[k][d];
62 
63  means[d] += val / double(nb_patterns);
64  means_of_squares[d] += Sqr(val) / double(nb_patterns);
65  }
66 
67  // Check means of squared data
68  status = (means_of_squares[0] == means_of_squares[0]);
69 
70  for (size_t d = 1; d < dimension; ++d)
71  status &= (means_of_squares[d] == means_of_squares[d]);
72 
73  std::vector< std::vector<double> > transformed_data(nb_patterns, std::vector<double>(dimension));
74 
75  if (status) // Means for squared data are in legal range
76  {
77  // Deviation computed with König-Huygens formula
78  for (size_t d = 0; d < dimension; ++d)
79  deviations[d] = sqrt(means_of_squares[d] - Sqr(means[d]));
80 
81  // Data patterns centered and reduced
82  for (size_t d = 0; d < dimension; ++d)
83  {
84  double mu = means[d];
85  double sigma = deviations[d];
86 
87  if (sigma != 1.0)
88  for (size_t k = 0; k < nb_patterns; ++k)
89  transformed_data[k][d] = (data[k][d] - mu) / sigma;
90  else
91  for (size_t k = 0; k < nb_patterns; ++k)
92  transformed_data[k][d] = data[k][d] - mu;
93  }
94  }
95  else
96  {
97  // Deviation computed in "classical" way
98  // Data patterns centered simultameously
99  for (size_t d = 0; d < dimension; ++d)
100  {
101  double mu = means[d];
102  double sigma = 0.0;
103 
104  for (size_t k = 0; k < nb_patterns; ++k)
105  {
106  double val = data[k][d];
107 
108  transformed_data[k][d] = val - mu;
109  sigma += Sqr(val - mu) / double(nb_patterns);
110  }
111

113  deviations[d] = sigma;
114

122

490 std::multimap<double, MatrixDouble> PCA::makeCorrelationSpectralEigensystem(double g) const
491 {
492  std::multimap<double, MatrixDouble> eigen_pairs;
493 
494  double delta = 4 * g * g;
495 
496  double m_1 = sqrt(delta) / (2 * g);
497 
498  double fraction = 1.0 / sqrt(1.0 + m_1 * m_1);
499 
500  MatrixDouble eigen_vector_1((size_t)2, (size_t)1);
501  MatrixDouble eigen_vector_2((size_t)2, (size_t)1);
502 
503  eigen_vector_1.At(0, 0) = fraction;
504  eigen_vector_1.At(1, 0) = m_1 * fraction;
505 
506  eigen_vector_2.At(0, 0) = - m_1 * fraction;
507  eigen_vector_2.At(1, 0) = fraction;
508 
509  double eigen_value_1 = (2 + sqrt(delta)) / 2.0;
510  double eigen_value_2 = (2 - sqrt(delta)) / 2.0;
511 
512  eigen_pairs.insert(std::make_pair(eigen_value_1, eigen_vector_1));
513  eigen_pairs.insert(std::make_pair(eigen_value_2, eigen_vector_2));
514 
515  return eigen_pairs;
516 }

664 std::vector<std::vector<double>> PCA::ReverseTransform(const std::vector< std::vector<double> > &data) const
665 {
666  size_t nb_patterns = data.size();
667  std::vector< std::vector<double> > new_patterns;
668  SquareMatrixDouble mat(dimension, 0.0);
669 
670  size_t j = 0;
671 
672  for (auto rev_iter = eigensystem.rbegin(); rev_iter != eigensystem.rend(); ++rev_iter)
673  {
674  MatrixDouble vect = rev_iter->second;
675 
676  for (size_t i = 0; i < dimension; ++i)
677  mat[i][j] = vect[i][0];
678 
679  ++j;
680  }
681 
682  for (size_t p = 0; p < nb_patterns; ++p)
683  {
684  std::vector<double> pattern = data[p];
685  std::vector<double> new_pattern(dimension);
686

688  {
689  double cumul = 0.0;
690 
691  for (size_t k = 0; k < dimension; ++k)
692  cumul += mat[i][k] * pattern[k];
693 
694  new_pattern[i] = cumul + means[i];
695  }
696 
697  new_patterns.push_back(new_pattern);
698  }
699 
700  return new_patterns;
701 }
