153 void Map::Remove(iterator first, iterator end_)
154 {
155  // cannot begin with end()
156  if (first == end())
157  throw ExceptionDomain(_("First iterator is end()."));
158  // null range
159  if (first == end_)
160  throw ExceptionInvalidArgument(_("First and end iterators are equal."));
161  // if end is end()
162  if (end_ == end())
163  {
164  data.erase(first, end_);
165  }
166  // if end is not the end() of the map, check that it is really after first
167  for (auto tmp = first; tmp != end(); ++tmp)
168  {
169  if (tmp == end_)
170  {
171  data.erase(first, end_);
172  return;
173  }
174  }
175  throw ExceptionInvalidArgument(_("End iterator is before first."));
176 }

188 void Map::Deserialize(xml::Element &el)
189 {
190  if (el.GetValue() != "Map")

193  _("Wrong XML element."));

205  {
206  // XXX throw?

208  String(_("Cannot deserialize: ")) + te.GetValue() + String(_(" because ")) + e.what());
209  }
