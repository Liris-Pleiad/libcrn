140 void Vector::ReorderFrom(const std::vector<size_t> &from)
141 {
142  if (from.size() != data.size())

145  _("bad changeset size."));

150  cont.insert(i);

268 void Vector::Deserialize(xml::Element &el)
269 {
270  if (el.GetValue() != getClassName())
271  {
272  throw ExceptionInvalidArgument(StringUTF8("void Vector::Deserialize(xml::Element &el): ") +
273  _("Wrong XML element."));
274  }
275  std::multimap<int, SObject> xmllist;
276  for (xml::Element te = el.BeginElement(); te != el.EndElement(); ++te)
277  {
278  SObject d(nullptr);
279  try
280  {
281  d = DataFactory::CreateData(te);
282  }
283  catch (crn::Exception &)
284  { // XXX throw?
285  CRNWarning(String(U"void Vector::Deserialize"
286  U"(xml::Element &el): ") + String(_("Unknown XML element: ")) +
287  te.GetValue());
288  }
289  if (!d)
290  continue;
291  int index;
292  try { index = te.GetAttribute<int>("vector_index", false); } catch (...) { index = int(xmllist.size()); }
293  xmllist.insert(std::make_pair(index, d));
294  }
295  Clear();
296  for (auto & elem : xmllist)
297  data.push_back(elem.second);
298  ShrinkToFit();
299 }
