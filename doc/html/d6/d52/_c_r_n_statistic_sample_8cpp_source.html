39 double crn::Mean(const std::vector<double> &v)
40 {
41  const auto s = double(v.size());
42  auto m = std::accumulate(v.begin(), v.end(), 0.0);
43 
44  if (!std::isinf(m))
45  return m / s;
46  else
47  {
48  m = 0.0;
49  for (auto x : v)
50  m += x / s;

59 std::vector<double> crn::MeanPattern(const std::vector<std::vector<double>> &m)
60 {
61  auto mp = std::vector<double>(m.front().size(), 0.0);
62 
63  for (const auto &row : m)
64  for (auto k : Range(row))
65  mp[k] += row[k];
66 
67  auto valid = true;
68  for (auto k : Range(mp))
69  if (std::isinf(mp[k]))
70  {
71  valid = false;
72  break;
73  }
74

76  std::transform(mp.begin(), mp.end(), mp.begin(), std::bind(std::divides<double>{}, _1, double(m.size())));
77  else
78  {
79  auto s = double(m.size());
80  std::vector<double>(m.front().size(), 0.0).swap(mp);
81 
82  for (const auto &row : m)
83  for (auto k : Range(row))
84  mp[k] += row[k] / s;
85  }
86 
87  return mp;

103 double crn::Variance(const std::vector<double> &v)
104 {
105  const auto m = Mean(v);
106  auto var = 0.0;
107  const auto s = double(v.size());
108 
109  for (double d : v)
110  var += Sqr(d - m);
111 
112  if (!std::isinf(var))
113  var /= s;
114  else
115  {
116  var = 0.0;
117  for (double d : v)
118  var += Sqr(d - m) / s;
119  }
120 
121  return var;
122 }

128 std::vector<std::vector<double>> crn::MakeCovariance(const std::vector<std::vector<double>> &m)
129 {
130  const auto card = m.size();
131  const auto dim = m.front().size();
132  auto cov = std::vector<std::vector<double>>(dim, std::vector<double>(dim));
133 
134  for (size_t i = 0; i < dim; ++i)
135  for (auto j = i; j < dim; ++j)
136  {
137  for (size_t k = 0; k < card; ++k)
138  cov[i][j] += m[k][i] * m[k][j];
139  cov[i][j] /= double(card);
140 
141  if (j != i)
142  cov[j][i] = cov[i][j];
143  }
144  return cov;
145 }

151 std::tuple<double, double, double> crn::MeanVarDev(const std::vector<double> &v)
152 {
153  // Compute cumul and cumul of squares in a same loop
154  auto m = 0.0;
155  auto m_2 = 0.0;
156  const auto s = double(v.size());
157 
158  for (double x : v)
159  {
160  m += x;
161  m_2 += Sqr(x);
162  }
163 
164  if (!std::isinf(m))
165  {
166  m /= s;
167  m_2 /= s;
168  }
169  else // If cumul too large, re-compute mean with values downscaled by cardinal
170  {
171  m = 0.0;
172  m_2 = 0.0;
173 
174  for (double x : v)
175  {
176  m += x / s;
177  m_2 += Sqr(x) / s;
178  }
179  }
180 
181  if (!std::isinf(m_2))
182  {
183  auto var = m_2 - Sqr(m);
184  return std::make_tuple(m, var, sqrt(var));
185  }
186  else // Konig-Huygens formula for variance cannot be used (too large values in cumul of squares)
187  {
188  auto var = 0.0;
189  for (double x : v)
190  var += Sqr(x - m);
191 
192  if (!std::isinf(var))
193  var /= s;
194  else
195  {
196  var = 0.0;
197  for (double x : v)
198  var += Sqr(x - m) / s;
199  }
200  return std::make_tuple(m, var, sqrt(var));
201  }
202 }

215 std::vector<double> crn::Quantiles(const std::vector<double> &v, size_t q, bool sort_flag)
216 {
217  if ((q < 3) || (q > v.size()))
218  throw ExceptionDomain(StringUTF8("StatisticSample::Quantiles(const std::vector<double>&, size_t, bool): ") + _("Illegal range."));
219 
220  auto qt = std::vector<double>(q - 1);
221  const auto jump = v.size() / q;
222  auto index = jump;
223 
224  if (sort_flag)
225  {
226  auto w = v;
227  std::sort(w.begin(), w.end());
228  for(size_t k = 0; k < q - 1; ++k)
229  {
230  qt[k] = w[index];
231  index += jump;
232  }
233  }
234  else
235  for(size_t k = 0; k < q - 1; ++k)
236  {
237  qt[k] = v[index];
238  index += jump;
239  }
240 
241  return qt;
242 }

249 Histogram crn::MakeHistogram(const std::vector<double> &v, size_t nb_bins)
250 {
251  auto mM = MinMax(v);
252  auto left_bound = std::get<0>(mM);
253  auto right_bound = std::get<1>(mM);
254  auto range = right_bound - left_bound;
255  auto delta = range / double(nb_bins);
256  auto h = Histogram(nb_bins);
257  if (nb_bins == 1) // trivial case
258  h.SetBin(0, (unsigned int)v.size());
259  else
260  for (auto d : v)
261  {
262  auto id = size_t((d - left_bound) / delta);
263  if (id >= nb_bins)
264  id = nb_bins - 1;
265  h.IncBin(id);
266  }
267  return h;
268 }
