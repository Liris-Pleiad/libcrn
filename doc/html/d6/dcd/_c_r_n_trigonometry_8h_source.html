377  {

396  template<typename ITER> ITER AngularMedian(ITER beg, ITER en)
397  {
398  if (beg == en)
399  throw ExceptionDomain("AngularMedian(): empty set of angles.");
400  auto mdist = std::vector<double>(std::distance(beg, en), 0.0);
401  auto cnt1 = size_t(0);
402  for (auto it1 = beg; it1 != en; ++cnt1, ++it1)
403  {
404  auto cnt2 = cnt1 + 1;
405  for (auto it2 = std::next(it1); it2 != en; ++cnt2, ++it2)
406  {
407  const auto d = AngularDistance(*it1, *it2);
408  mdist[cnt1] += d;
409  mdist[cnt2] += d;
410  }
411  }
412  return std::next(beg, std::min_element(mdist.begin(), mdist.end()) - mdist.begin());
413  }
