28 static std::vector<int> local_max_cube(const std::vector<long> &cube)
29 {
30  const auto size = cube.size();
31  size_t GOFFSET = size;
32  size_t BOFFSET = GOFFSET * size;
33 
34  // find local max
35  auto mcube = std::vector<int>(size * size * size, 0);
36  for (size_t b = 0; b < size; b++)
37  for (size_t g = 0; g < size; g++)
38  for (size_t r = 0; r < size; r++)
39  { // for each value in the cube
40  long val = cube[r + g * GOFFSET + b * BOFFSET];
41  for (int tb = Cap((int)b - 2, 0, (int)size - 1); tb <= Cap((int)b + 2, 0, (int)size - 1); tb++)
42  for (int tg = Cap((int)g - 2, 0, (int)size - 1); tg <= Cap((int)g + 2, 0, (int)size - 1); tg++)
43  for (int tr = Cap((int)r - 2, 0, (int)size - 1); tr <= Cap((int)r + 2, 0, (int)size - 1); tr++)
44  { // for each neighbor
45  if ((tr == int(r)) && (tg == int(g)) && (tb == int(b)))
46  continue;
47  if (cube[tr + tg * GOFFSET + tb * BOFFSET] > val)
48  {
49  mcube[r + g * GOFFSET + b * BOFFSET] = -1;
50  }
51  }
52  }
53  bool mod = false;
54  do
55  {
56  mod = false;
57  for (size_t b = 0; b < size; b++)
58  for (size_t g = 0; g < size; g++)
59  for (size_t r = 0; r < size; r++)
60  { // for each value in the cube
61  if (mcube[r + g * GOFFSET + b * BOFFSET] == -1)
62  continue;
63  long val = cube[r + g * GOFFSET + b * BOFFSET];
64  for (int tb = Cap((int)b - 1, 0, (int)size - 1); tb <= Cap((int)b + 1, 0, (int)size - 1); tb++)
65  for (int tg = Cap((int)g - 1, 0, (int)size - 1); tg <= Cap((int)g + 1, 0, (int)size - 1); tg++)
66  for (int tr = Cap((int)r - 1, 0, (int)size - 1); tr <= Cap((int)r + 1, 0, (int)size - 1); tr++)
67  { // for each neighbor
68  if ((tr == int(r)) && (tg == int(g)) && (tb == int(b)))
69  continue;
70  if ((cube[tr + tg * GOFFSET + tb * BOFFSET] == val) &&
71  (mcube[tr + tg * GOFFSET + tb * BOFFSET] == -1))
72  {
73  mcube[r + g * GOFFSET + b * BOFFSET] = -1;
74  mod = true;
75  }
76  }
77  }
78  } while (mod);
79 
80  return mcube;
81 }

89 ColorModel::ColorModel(const ImageRGB &img, size_t max)
90 {
91  const size_t CUBESIZE = 256;
92  const size_t GOFFSET = CUBESIZE;
93  const size_t BOFFSET = GOFFSET * CUBESIZE;
94 
95  // extract colorcube
96  auto cube = std::vector<long>(CUBESIZE * CUBESIZE * CUBESIZE, 0);
97  for (size_t tmp = 0; tmp < img.Size(); tmp++)
98  {
99  cube[img.At(tmp).r + img.At(tmp).g * GOFFSET + img.At(tmp).b * BOFFSET] += 1;
100  }
101  // search through scales
102  std::vector<pixel::RGB8> colors;
103  size_t bg = 0;
104  for (size_t frac = 2; frac < 7; frac++)
105  {
106  // create scaled cube
107  size_t ssize = CUBESIZE >> frac;
108  size_t sGOFFSET = ssize;
109  size_t sBOFFSET = sGOFFSET * ssize;
110  auto scube = std::vector<long>(sGOFFSET * sBOFFSET, 0);
111  for (size_t b = 0; b < CUBESIZE; b++)
112  for (size_t g = 0; g < CUBESIZE; g++)
113  for (size_t r = 0; r < CUBESIZE; r++)
114  scube[(r >> frac) + (g >> frac) * sGOFFSET +
115  (b >> frac) * sBOFFSET] += cube[r + g * GOFFSET + b * BOFFSET];
116 
117  auto mcube = local_max_cube(scube);
118 
119  // look for modes
120  bg = 0;
121  long bgcnt = 0;
122  for (size_t b = 0; b < ssize; b++)
123  for (size_t g = 0; g < ssize; g++)
124  for (size_t r = 0; r < ssize; r++)
125  if (mcube[r + g * sGOFFSET + b * sBOFFSET] == 0)
126  { // found a max
127  // compute color
128  long cnt = 0;
129  long ar = 0, ag = 0, ab = 0;
130  for (size_t tb = b << frac; tb < (b + 1) << frac; tb++)
131  for (size_t tg = g << frac; tg < (g + 1) << frac; tg++)
132  for (size_t tr = r << frac; tr < (r + 1) << frac; tr++)
133  {
134  cnt += cube[tr + tg * GOFFSET + tb * BOFFSET];
135  ar += long(tr) * cube[tr + tg * GOFFSET + tb * BOFFSET];
136  ag += long(tg) * cube[tr + tg * GOFFSET + tb * BOFFSET];
137  ab += long(tb) * cube[tr + tg * GOFFSET + tb * BOFFSET];
138  }
139  // store color
140  colors.emplace_back(uint8_t(ar / cnt), uint8_t(ag / cnt), uint8_t(ab / cnt));
141  if (cnt > bgcnt)
142  {
143  bg = colors.size() - 1;
144  bgcnt = cnt;
145  }
146  }
147 
148  if (colors.size() < max)
149  {
150  break;
151  }
152 
153  colors.clear();
154  }
155  const auto &bgcol = colors[bg];
156  std::sort(colors.begin(), colors.end(),
157  [](const pixel::RGB8 &p1, const pixel::RGB8 &p2)
158  { return (p1.r + p1.g + p1.b) < (p2.r + p2.g + p2.b); }
159  );
160  bg = std::find(colors.begin(), colors.end(), bgcol) - colors.begin();
161 
162  for (size_t tmp = 0; tmp < colors.size(); tmp++)
163  {
164  if (tmp == 0)
165  ink = colors[tmp];
166  else if (tmp == bg)
167  paper = colors[tmp];
168  else
169  other.push_back(colors[tmp]);
170  }
171 }
