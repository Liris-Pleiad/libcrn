30 double LinearInterpolation::operator[](double x) const
31 {
32  double slope, x0, y0;
33  if (x < data.front().X)
34  {
35  x0 = data[0].X;
36  y0 = data[0].Y;
37  slope = (data[1].Y - y0) / (data[1].X - x0);
38  }
39  if (x > data.back().X)
40  {
41  x0 = data.back().X;
42  y0 = data.back().Y;
43  slope = (data[data.size() - 2].Y - y0) / (data[data.size() - 2].X - x0);
44  }
45  else
46  {
47  for (size_t tmp = 0; tmp < data.size() - 1; ++tmp)
48  {
49  if (data[tmp].X == x)
50  return data[tmp].Y;
51  else if ((data[tmp].X <= x) && (data[tmp + 1].X >= x))
52  {
53  x0 = data[tmp].X;
54  y0 = data[tmp].Y;
55  slope = (data[tmp + 1].Y - y0) / (data[tmp + 1].X - x0);
56  break;
57  }
58  }
59  }
60  return y0 + slope * (x - x0);
61 }
