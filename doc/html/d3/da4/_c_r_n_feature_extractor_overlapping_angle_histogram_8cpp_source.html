34 static inline size_t chooseangle(const crn::Angle<ByteAngle> &a, const std::vector<crn::Angle<ByteAngle> > &angles)
35 {
36  std::vector<uint8_t> distances(angles.size());
37  for (size_t tmp = 0; tmp < angles.size(); ++tmp)
38  distances[tmp] = crn::AngularDistance(a, angles[tmp]);
39  return std::min_element(distances.begin(), distances.end()) - distances.begin();
40 }

49 SObject FeatureExtractorOverlappingAngleHistogram::Extract(Block &b)
50 {
51  const size_t ncenters = 9;
52  // find center and std deviation
53  uint64_t xsum = 0, ysum = 0, x2sum = 0, y2sum = 0, vsum = 0;
54  SImageGray ig(b.GetGray());
55  FOREACHPIXEL(x, y, *ig)
56  {
57  int val = 255 - ig->At(x, y);
58  vsum += val;
59  xsum += x * val;
60  ysum += y * val;
61  x2sum += x * x * val;
62  y2sum += y * y * val;
63  }
64  if (!xsum) // white image
65  return std::make_shared<MatrixDouble>(ncenters * nbangles, 1, 0.0);
66  int xbar;
67  int ybar;
68  double xvar;
69  double yvar;
70  if (vsum == 0)
71  {
72  xbar = int(ig->GetWidth()) / 2;
73  ybar = int(ig->GetHeight()) / 2;
74  xvar = int(ig->GetWidth()) / 6;
75  yvar = int(ig->GetHeight()) / 6;
76  }
77  else
78  {
79  xbar = int(xsum / vsum);
80  ybar = int(ysum / vsum);
81  xvar = (double(x2sum) - double(xsum * xsum) / double(vsum)) / double(vsum);
82  yvar = (double(y2sum) - double(ysum * ysum) / double(vsum)) / double(vsum);
83  }
84 
85  // compute centers (sqrt(1.4×var) is the distance where the Gaussian evaluates to 0.5)
86  std::vector<crn::Point2DInt> centers(ncenters);
87  centers[0] = crn::Point2DInt(xbar, ybar);
88  centers[1] = crn::Point2DInt(xbar - int(sqrt(distance * xvar)), ybar);
89  centers[2] = crn::Point2DInt(xbar - int(sqrt(distance * xvar)), ybar - int(sqrt(distance * yvar)));
90  centers[3] = crn::Point2DInt(xbar, ybar - int(sqrt(distance * yvar)));
91  centers[4] = crn::Point2DInt(xbar + int(sqrt(distance * xvar)), ybar - int(sqrt(distance * yvar)));
92  centers[5] = crn::Point2DInt(xbar + int(sqrt(distance * xvar)), ybar);
93  centers[6] = crn::Point2DInt(xbar + int(sqrt(distance * xvar)), ybar + int(sqrt(distance * yvar)));
94  centers[7] = crn::Point2DInt(xbar, ybar + int(sqrt(distance * yvar)));
95  centers[8] = crn::Point2DInt(xbar - int(sqrt(distance * xvar)), ybar + int(sqrt(distance * yvar)));
96 
97  // precompute Gaussian
98  int maxx = int(2*ig->GetWidth());
99  int maxy = int(2*ig->GetHeight());
100  std::vector<std::vector<double> > gauss(maxx, std::vector<double>(maxy));
101  const double sigmax = 2.0 * xvar / 4.0;
102  const double sigmay = 2.0 * yvar / 4.0;
103  for (int x = 0; x < maxx; ++x)
104  for (int y = 0; y < maxy; ++y)
105  gauss[x][y] = exp(-((x * x) + (y * y)) / (sigmax * sigmay));
106 
107  // compute features
108  SImageGradient igr(b.GetGradient());
109  int cnt = 0;
110  std::vector<double> feats(ncenters * nbangles, 0);
111  FOREACHPIXEL(x, y, *igr)
112  {
113  if (igr->IsSignificant(x, y))
114  {
115  cnt += 1;
116  size_t angle = chooseangle(igr->At(x, y).theta, angletab);
117  for (size_t tmp = 0; tmp < ncenters; ++tmp)
118  {
119  feats[tmp * nbangles + angle] += gauss[crn::Abs(int(x) - centers[tmp].X)][crn::Abs(int(y) - centers[tmp].Y)];
120  }
121  }
122  }
123  if (cnt != 0)
124  for (auto & feat : feats)
125  feat /= cnt;
126 
127  return std::make_shared<MatrixDouble>(feats);
128 }
