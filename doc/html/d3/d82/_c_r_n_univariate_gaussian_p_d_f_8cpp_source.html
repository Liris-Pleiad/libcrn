111 double UnivariateGaussianPDF::IntersectionAbscissa(const UnivariateGaussianPDF &pdf) const
112 {
113  double mu1 = GetMean();
114  double mu2 = pdf.GetMean();
115  double sigma1 = GetDeviation();
116  double sigma2 = pdf.GetDeviation();
117  double mu1_2 = mu1 * mu1;
118  double mu2_2 = mu2 * mu2;
119  double sigma1_2 = sigma1 * sigma1;
120  double sigma2_2 = sigma2 * sigma2;
121 
122  double a = sigma1_2 - sigma2_2;
123  double b = 2 * (mu1 * sigma2_2 - mu2 * sigma1_2);
124  double c = mu2_2 * sigma1_2 - mu1_2 * sigma2_2 - 2 * sigma1_2 * sigma2_2 * (log(sigma1) - log(sigma2));
125 
126  std::set<double> roots = crn::QuadraticEquation::RealRoots(a, b, c);
127 
128  if (roots.size() == 1)
129  {
130  return *roots.begin();
131  }
132  else if (roots.size() == 2)
133  {
134  double x = *roots.begin();
135 
136  if((x - mu1) * (mu2 - x) < 0.0)
137  {
138  x = *(++roots.begin());
139  }
140 
141  return x;
142  }
143 
144  return std::numeric_limits<double>::infinity();
145 }

154 double UnivariateGaussianPDF::IntersectionAbscissaApprox(const UnivariateGaussianPDF &pdf) const
155 {
156  double mean = GetMean();
157  double ext_mean = pdf.GetMean();
158 
159  if (mean != ext_mean)
160  {
161  double left_bound = Min(mean, ext_mean);
162  double right_bound = Max(mean, ext_mean);
163  double diff = ValueAt(left_bound) - pdf.ValueAt(left_bound);
164  double unsigned_diff = fabs(diff);
165  double ref_sgn = diff / unsigned_diff;
166 
167  while (right_bound - left_bound > std::numeric_limits<double>::epsilon())
168  {
169  double middle = (right_bound + left_bound) / 2;
170 
171  diff = ValueAt(middle) - pdf.ValueAt(middle);
172  unsigned_diff = fabs(diff);
173 
174  if ((diff / unsigned_diff) == ref_sgn)
175  {
176  left_bound = middle;
177  }
178  else
179  {
180  right_bound = middle;
181  }
182  }
183 
184  return (right_bound + left_bound) / 2;
185  }
186 
187  return std::numeric_limits<double>::infinity();
188 }

199 std::vector<double> UnivariateGaussianPDF::MakeRandomSample(size_t n, size_t m, bool reseed) const
200 {
201  if (reseed)
202  {
203  srand((unsigned int)time(nullptr));
204  }
205 
206  std::vector<double> sample;
207  double sigma = sqrt(variance);
208 
209  for (size_t k = 0; k < n; k++)
210  {
211  // cumulate a uniform sample with m numbers
212 
213  double s = 0.0;
214 
215  for (size_t i = 0; i < m; i++)
216  {
217  s += ((double)(rand())) / ((double)(RAND_MAX));
218  }
219 
220  // Gaussian random number from uniform distribution
221 
222  s -= double(m) / 2.0;
223  s *= sigma * sqrt(12.0 / double(m));
224  s += mean;
225 
226  sample.push_back(s);
227  }
228 
229  return sample;
230 }
