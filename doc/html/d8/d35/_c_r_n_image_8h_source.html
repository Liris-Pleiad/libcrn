327  template<typename IMG, typename T> IMG Downgrade(const Image<T> &img)
328  {
329  using new_pixel = typename IMG::pixel_type;
330  using common_pixel = typename std::common_type<T, new_pixel>::type;
331 
332  auto res = IMG(img.GetWidth(), img.GetHeight());
333  const auto mM = MinMax(img);
334  if ((common_pixel(mM.first) < common_pixel(std::numeric_limits<new_pixel>::lowest())) ||
335  (common_pixel(mM.second) > common_pixel(std::numeric_limits<new_pixel>::max())))
336  {
337  for (auto tmp : Range(img))
338  {
339  auto val = common_pixel(img.At(tmp)) - common_pixel(mM.first);
340  val = (val * common_pixel(std::numeric_limits<new_pixel>::max())) / common_pixel(mM.second - mM.first);
341  res.At(tmp) = new_pixel(val);
342  }
343  }
344  else
345  {
346  for (auto tmp : Range(img))
347  res.At(tmp) = new_pixel(img.At(tmp));
348  }
349  return res;
350  }

357  template<typename IMG, typename T> IMG Downgrade(const Image<pixel::RGB<T>> &img)
358  {
359  using new_pixel = typename IMG::pixel_type;
360  using new_base = typename new_pixel::type;
361  using common_base = typename std::common_type<T, typename new_pixel::type>::type;
362  using common_pixel = pixel::RGB<common_base>;
363 
364  auto res = IMG(img.GetWidth(), img.GetHeight());
365  auto m = std::numeric_limits<T>::max();
366  auto M = std::numeric_limits<T>::lowest();
367  for (const auto &p : img)
368  {
369  if (p.r < m) m = p.r;
370  if (p.g < m) m = p.g;
371  if (p.b < m) m = p.b;
372  if (p.r > M) M = p.r;
373  if (p.g > M) M = p.g;
374  if (p.b > M) M = p.b;
375  }
376  if ((common_base(m) < common_base(std::numeric_limits<new_base>::lowest())) ||
377  (common_base(M) > common_base(std::numeric_limits<new_base>::max())))
378  {
379  for (auto tmp : Range(img))
380  {
381  auto val = common_pixel(img.At(tmp)) - common_pixel(m);
382  val = (val * common_base(std::numeric_limits<new_base>::max())) / common_base(M - m);
383  res.At(tmp) = new_pixel(val);
384  }
385  }
386  else
387  {
388  for (auto tmp : Range(img))
389  res.At(tmp) = new_pixel(img.At(tmp));
390  }
391  return res;
392  }
